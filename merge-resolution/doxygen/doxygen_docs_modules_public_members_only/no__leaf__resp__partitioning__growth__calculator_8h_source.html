140{
141 double net_assimilation_rate_leaf{0.0};
142 double net_assimilation_rate_stem{0.0};
143 double net_assimilation_rate_root{0.0};
144 double net_assimilation_rate_rhizome{0.0};
145 double net_assimilation_rate_grain{0.0};
146
147 // Determine the carbon flux to use for the non-leaf organs
148 double nonleaf_carbon_flux;
149 if (canopy_assimilation_rate < 0) {
150 nonleaf_carbon_flux = 0.0;
151 } else {
152 nonleaf_carbon_flux = canopy_assimilation_rate;
153 }
154
155 // Calculate the rate of new leaf production
156 if (kLeaf > 0) {
157 if (canopy_assimilation_rate < 0) {
158 // Assimilation is negative here, so this removes leaf mass
159 net_assimilation_rate_leaf = canopy_assimilation_rate;
160 } else {
161 net_assimilation_rate_leaf = canopy_assimilation_rate * kLeaf;
162 }
163 } else {
164 net_assimilation_rate_leaf = 0.0;
165 }
166
167 // Calculate the rate of new stem production
168 if (kStem >= 0) {
169 net_assimilation_rate_stem = nonleaf_carbon_flux * kStem;
170 net_assimilation_rate_stem = resp(net_assimilation_rate_stem, mrc1, temp);
171 } else {
172 net_assimilation_rate_stem = 0.0;
173 }
174
175 // Calculate the rate of new root production
176 if (kRoot > 0) {
177 net_assimilation_rate_root = nonleaf_carbon_flux * kRoot;
178 net_assimilation_rate_root = resp(net_assimilation_rate_root, mrc2, temp);
179 } else {
180 net_assimilation_rate_root = 0.0;
181 }
182
183 // Calculate the rate of new rhizome production
184 if (kRhizome > 0) {
185 net_assimilation_rate_rhizome = nonleaf_carbon_flux * kRhizome;
186 net_assimilation_rate_rhizome = resp(net_assimilation_rate_rhizome, mrc2, temp);
187 } else {
188 net_assimilation_rate_rhizome = 0.0;
189 }
190
191 // Calculate the rate of grain production
192 if (kGrain > 0) {
193 net_assimilation_rate_grain = nonleaf_carbon_flux * kGrain;
194 } else {
195 net_assimilation_rate_grain = 0.0;
196 }
197
198 // Update the output quantity list
199 update(net_assimilation_rate_leaf_op, net_assimilation_rate_leaf);
200 update(net_assimilation_rate_stem_op, net_assimilation_rate_stem);
201 update(net_assimilation_rate_root_op, net_assimilation_rate_root);
202 update(net_assimilation_rate_rhizome_op, net_assimilation_rate_rhizome);
