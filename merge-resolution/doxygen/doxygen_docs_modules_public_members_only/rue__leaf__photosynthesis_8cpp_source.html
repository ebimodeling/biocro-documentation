17// Determine assimilation, stomatal conductance, and Ci following `c3photoC()`.
18// Here, rather than using the FvCB model for C3 photosynthesis, gross
19// assimilation is determined using a simple RUE model. Water stress is ignored.
20// Respiration and stomatal conductance are otherwise calculated using the same
21// methods as `c3photoC()`.
22photosynthesis_outputs rue_photo(
23 double Qp, // mol / m^2 / s
24 double alpha_rue, // dimensionless
25 double Tleaf, // degrees C
26 double RH, // dimensionless from Pa / Pa
27 double Rd0, // mol / m^2 / s
28 double bb0, // mol / m^2 / s
29 double bb1, // dimensionless
30 double Ca, // dimensionless from mol / mol
31 double gbw // mol / m^2 / s
32)
33{
34 // Convert leaf temperature to Kelvin
35 double const tlk = Tleaf + celsius_to_kelvin; // K
36
37 // Calculate gross assimilation based on incident light and radiation use
38 // efficiency
39 double const ag = Qp * alpha_rue; // mol / m^2 / s
40
41 // Determine respiration from the leaf temperature using Arrhenius exponents
42 // from Bernacchi et al. (2001) Plant, Cell and Environment, 24(2), 253-259.
43 // https://doi.org/10.1111/j.1365-3040.2001.00668.x
44 double const rd = Rd0 *
45 arrhenius_exponential(
46 18.72,
47 46.39e3,
48 tlk); // mol / m^2 / s
49
50 // Determine net assimilation
51 double an = ag - rd; // mol / m^2 / s
52
53 // The net CO2 assimilation is the smaller of the RUE-limited and
54 // conductance-limited rates. This will prevent the calculated Ci and Cs
55 // values from ever being < 0. This seems to be an important restriction to
56 // prevent numerical errors while running the Ball-Berry model. Here we
57 // just use a very large value for gsw; this may cause a slight understimate
58 // of the net assimilation rate in some cases.
59 double const an_conductance =
60 conductance_limited_assim(Ca * 1e6, gbw, 1e3) * 1e-6; // mol / m^2 / s
61
62 an = std::min(an_conductance, an); // mol / m^2 / s
63
64 // Determine stomatal conductance using the Ball-Berry model, equating the
65 // leaf and air temperatures (mol / m^2 / s)
66 stomata_outputs const BB_res =
67 ball_berry_gs(an, Ca, RH, bb0, bb1, gbw, Tleaf, Tleaf);
68
69 double const gs = BB_res.gsw * 1e-3;
70
71 // Determine intercellular CO2 concentration
72 double const ci = Ca - an * (dr_boundary / gbw + dr_stomata / gs); // dimensionless
73
74 return photosynthesis_outputs{
75 .Assim = an * 1e6, // micromol / m^2 / s
76 .Assim_conductance = an_conductance * 1e6, // micromol / m^2 / s
77 .Ci = ci * 1e6, // micromol / mol
78 .GrossAssim = ag * 1e6, // micromol / m^2 / s
79 .Gs = gs * 1e3, // mmol / m^2 / s
80 .Cs = BB_res.cs, // micromol / m^2 / s
81 .RHs = BB_res.hs, // dimensionless from Pa / Pa
82 .Rp = 0.0 // micromol / m^2 / s
83 };
84}

122void rue_leaf_photosynthesis::do_operation() const
123{
124 // Make an initial guess for boundary layer conductance
125 double const gbw_guess{1.2}; // mol / m^2 / s
126
127 // Get an initial estimate of stomatal conductance, assuming the leaf is at
128 // air temperature
129 const double initial_stomatal_conductance =
130 rue_photo(
131 incident_ppfd * 1e-6, // mol / m^2 / s
132 alpha_rue, // dimensionless
133 temp, // degrees C
134 rh, // dimensionless from Pa / Pa
135 Rd * 1e-6, // mol / m^2 / s
136 b0, // mol / m^2 / s
137 b1, // dimensionless
138 Catm * 1e-6, // dimensionless from mol / mol
139 gbw_guess // mol / m^2 / s
140 )
141 .Gs; // mmol / m^2 / s
142
143 // Calculate a new value for leaf temperature
144 const struct ET_Str et = c3EvapoTrans(
145 average_absorbed_shortwave,
146 temp,
147 rh,
148 windspeed,
149 height,
150 specific_heat_of_air,
151 initial_stomatal_conductance,
152 minimum_gbw,
153 windspeed_height);
154
155 const double leaf_temperature = temp + et.Deltat; // deg. C
156
157 // Calculate final values for assimilation, stomatal conductance, and Ci
158 // using the new leaf temperature
159 const photosynthesis_outputs photo =
160 rue_photo(
161 incident_ppfd * 1e-6, // mol / m^2 / s
162 alpha_rue, // dimensionless
163 leaf_temperature, // degrees C
164 rh, // dimensionless from Pa / Pa
165 Rd * 1e-6, // mol / m^2 / s
166 b0, // mol / m^2 / s
167 b1, // dimensionless
168 Catm * 1e-6, // dimensionless from mol / mol
169 et.boundary_layer_conductance // mol / m^2 / s
170 );
171
172 // Update the outputs
173 update(Assim_op, photo.Assim);
174 update(GrossAssim_op, photo.GrossAssim);
175 update(Rp_op, photo.Rp);
176 update(Ci_op, photo.Ci);
177 update(Gs_op, photo.Gs);
178 update(TransR_op, et.TransR);
179 update(EPenman_op, et.EPenman);
180 update(EPriestly_op, et.EPriestly);
181 update(leaf_temperature_op, leaf_temperature);
182 update(gbw_op, et.boundary_layer_conductance);
183}
