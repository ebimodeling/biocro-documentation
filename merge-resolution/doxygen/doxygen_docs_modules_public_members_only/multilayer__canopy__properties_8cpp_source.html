102void multilayer_canopy_properties::run() const
103{
104 // Calculate values of incident photosynthetically active photon flux
105 // density (PPFD) and absorbed shortwave energy throughout the canopy. Note
106 // that the `sunML` function expects input expects PPFD values, so we must
107 // convert photosynthetically active radiation (PAR) to PPFD using the
108 // energy content of light in the PAR band
109 struct Light_profile light_profile = sunML(
110 par_incident_direct / par_energy_content, // micromol / (m^2 beam) / s
111 par_incident_diffuse / par_energy_content, // micromol / m^2 / s
112 lai,
113 nlayers,
114 cosine_zenith_angle,
115 kd,
116 chil,
117 absorptivity_par,
118 heightf,
119 par_energy_content,
120 par_energy_fraction,
121 leaf_transmittance,
122 leaf_reflectance);
123
124 // Calculate windspeed throughout the canopy
125 double wind_speed_profile[nlayers];
126 WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile
127
128 // Calculate leaf nitrogen throughout the canopy
129 double leafN_profile[nlayers];
130 LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile
131
132 // Don't calculate anything based on the nitrogen profile
133 if (lnfun != 0) {
134 throw std::logic_error("Thrown by the multilayer_canopy_properties module: lnfun != 0 is not yet supported.");
135 }
136
137 // Update layer-dependent outputs
138 for (int i = 0; i < nlayers; ++i) {
139 update(sunlit_fraction_ops[i], light_profile.sunlit_fraction[i]);
140 update(sunlit_incident_ppfd_ops[i], light_profile.sunlit_incident_ppfd[i]);
141 update(sunlit_absorbed_ppfd_ops[i], light_profile.sunlit_absorbed_ppfd[i]);
142 update(sunlit_absorbed_shortwave_ops[i], light_profile.sunlit_absorbed_shortwave[i]);
143
144 update(shaded_fraction_ops[i], light_profile.shaded_fraction[i]);
145 update(shaded_incident_ppfd_ops[i], light_profile.shaded_incident_ppfd[i]);
146 update(shaded_absorbed_ppfd_ops[i], light_profile.shaded_absorbed_ppfd[i]);
147 update(shaded_absorbed_shortwave_ops[i], light_profile.shaded_absorbed_shortwave[i]);
148
149 update(average_incident_ppfd_ops[i], light_profile.average_incident_ppfd[i]);
150 update(average_absorbed_shortwave_ops[i], light_profile.average_absorbed_shortwave[i]);
151
152 update(incident_ppfd_scattered_ops[i], light_profile.incident_ppfd_scattered[i]);
153 update(height_ops[i], light_profile.height[i]);
154 update(windspeed_ops[i], wind_speed_profile[i]);
155 update(LeafN_ops[i], leafN_profile[i]);
156 }
157
158 // Update other outputs
159 update(canopy_direct_transmission_fraction_op, light_profile.canopy_direct_transmission_fraction);
160}
