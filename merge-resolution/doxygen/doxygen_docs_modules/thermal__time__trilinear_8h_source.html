148void thermal_time_trilinear::do_operation() const
149{
150 // Find the rate of change on a daily basis
151 double const rate_per_day = time < sowing_time ? 0.0
152 : temp <= tbase ? 0.0
153 : temp <= topt_lower ? temp - tbase
154 : temp <= topt_upper ? topt_lower - tbase
155 : temp <= tmax ? (tmax - temp) * (topt_lower - tbase) / (tmax - topt_upper)
156 : 0.0; // degrees C
157
158 // Convert to an hourly rate
159 double const rate_per_hour = rate_per_day / 24.0; // degrees C * day / hr
160
161 // Update the output quantity list
162 update(TTc_op, rate_per_hour);
163}
