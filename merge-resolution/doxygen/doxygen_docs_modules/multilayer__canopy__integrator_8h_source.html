152void multilayer_canopy_integrator::run() const
153{
154 double const LAIc = lai / nlayers;
155 double canopy_assimilation_rate{0.0};
156 double canopy_transpiration_rate{0.0};
157 double canopy_conductance{0.0};
158 double GrossAssim{0.0};
159 double canopy_photorespiration_rate{0.0};
160
161 // Integrate assimilation, transpiration, and conductance throughout the
162 // canopy
163 for (int i = 0; i < nlayers; ++i) {
164 double const sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
165 double const shaded_lai = *shaded_fraction_ips[i] * LAIc;
166
167 canopy_assimilation_rate += *sunlit_Assim_ips[i] * sunlit_lai +
168 *shaded_Assim_ips[i] * shaded_lai;
169
170 canopy_transpiration_rate += *sunlit_TransR_ips[i] * sunlit_lai +
171 *shaded_TransR_ips[i] * shaded_lai;
172
173 canopy_conductance += *sunlit_Gs_ips[i] * sunlit_lai +
174 *shaded_Gs_ips[i] * shaded_lai;
175
176 GrossAssim += *sunlit_GrossAssim_ips[i] * sunlit_lai +
177 *shaded_GrossAssim_ips[i] * shaded_lai;
178
179 canopy_photorespiration_rate += *sunlit_Rp_ips[i] * sunlit_lai +
180 *shaded_Rp_ips[i] * shaded_lai;
181 }
182
183 // Modify net assimilation to account for respiration
184 // Note: this was originally only done for the C3 canopy
185 // Note: it seems like this should not be necessary since the assimilation
186 // model includes respiration
187 canopy_assimilation_rate *= (1.0 - growth_respiration_fraction);
188
189 // For assimilation, we need to convert micromol / m^2 / s into
190 // Mg / ha / hr, assuming that all carbon is converted into biomass in the
191 // form of glucose (C6H12O6), i.e., six assimilated CO2 molecules contribute
192 // one glucose molecule. Using the molar mass of glucose in kg / mol, the
193 // conversion can be accomplished with the following factor:
194 // (1 glucose / 6 CO2) * (3600 s / hr) * (1e-6 mol / micromol) *
195 // (1e-3 Mg / kg) * (1e4 m^2 / ha)
196 // = 6e-3 s * mol * Mg * m^2 / (hr * micromol * kg * ha)
197 const double cf = physical_constants::molar_mass_of_glucose * 6e-3; // (Mg / ha / hr) / (micromol / m^2 / s)
198
199 // For transpiration, we need to convert mmol / m^2 / s into Mg / ha / hr
200 // using the molar mass of water in kg / mol, which can be accomplished by
201 // the following conversion factor:
202 // (3600 s / hr) * (1e-3 mol / mmol) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
203 // = 36 s * mol * Mg * m^2 / (hr * mmol * kg * ha)
204 const double cf2 = physical_constants::molar_mass_of_water * 36; // (Mg / ha / hr) / (mmol / m^2 / s)
205
206 update(canopy_assimilation_rate_op, canopy_assimilation_rate * cf);
207
208 update(GrossAssim_op, GrossAssim * cf);
209
210 update(canopy_assimilation_rate_op, canopy_assimilation_rate * cf);
211
212 update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
213
214 update(canopy_conductance_op, canopy_conductance);
215}
