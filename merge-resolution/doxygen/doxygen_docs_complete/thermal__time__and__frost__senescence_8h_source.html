252void thermal_time_and_frost_senescence::do_operation() const
253{
254 // Add the new tissue growth to the history vectors
255 assim_rate_stem_vec.push_back(net_assimilation_rate_stem);
256 assim_rate_root_vec.push_back(net_assimilation_rate_root);
257 assim_rate_rhizome_vec.push_back(net_assimilation_rate_rhizome);
258
259 // Initialize variables
260 double dLeafdeathrate{0.0};
261 double dLeaf{0.0};
262 double dStem{0.0};
263 double dRoot{0.0};
264 double dRhizome{0.0};
265 double dGrain{0.0};
266 double dLeafLitter{0.0};
267 double dStemLitter{0.0};
268 double dRootLitter{0.0};
269 double dRhizomeLitter{0.0};
270 double dstem_senescence_index{0.0};
271 double droot_senescence_index{0.0};
272 double drhizome_senescence_index{0.0};
273
274 // Calculate the leaf death rate due to frost
275 double frost_leaf_death_rate{0.0};
276 if (TTc >= seneLeaf) { // Leaf senescence has started
277 bool A = lat >= 0.0; // In Northern hemisphere
278 bool B = time >= 180.0; // In second half of the year
279 if ((A && B) || ((!A) && (!B))) { // Winter in either hemisphere
280 // frost_leaf_death_rate changes linearly from 100 to 0 as temp
281 // changes from Tfrostlow to Tfrosthigh and is limited to [0,100]
282 std::max(0.0, std::min(100.0, 100.0 * (Tfrosthigh - temp) /
283 (Tfrosthigh - Tfrostlow)));
284 }
285 }
286
287 // The current leaf death rate is the larger of the previously stored leaf
288 // death rate and the new frost death rate. I.e., the leaf death rate
289 // sometimes increases but never decreases
290 double current_leaf_death_rate = std::max(leafdeathrate,
291 frost_leaf_death_rate);
292
293 // Report the change in leaf death rate as the derivative of leafdeathrate.
294 // Note: this will probably only work well with the Euler method
295 dLeafdeathrate = current_leaf_death_rate - leafdeathrate;
296
297 // Calculate leaf senescence
298 if (TTc >= seneLeaf) {
299 // Use the leaf death rate to determine the change in leaf mass
300 double change = Leaf * current_leaf_death_rate * (0.01 / 24);
301
302 // Remobilize some of the lost leaf tissue and send the rest to the
303 // litter. EBL wonders: why does the leaf remobilize its own tissue?
304 dLeaf += -change + kLeaf * change * remobilization_fraction;
305 dLeafLitter += change * (1.0 - remobilization_fraction);
306 dRhizome += kRhizome * change * remobilization_fraction;
307 dStem += kStem * change * remobilization_fraction;
308 dRoot += kRoot * change * remobilization_fraction;
309 dGrain += kGrain * change * remobilization_fraction;
310 }
311
312 // Calculate stem senescence
313 if (TTc >= seneStem) {
314 // Look back in time to find out how much the tissue grew in the past.
315 double change = assim_rate_stem_vec.at(stem_senescence_index);
316
317 // Subtract the new growth from the tissue derivative
318 dStem -= change;
319
320 // Send the lost tissue to the litter
321 dStemLitter += change;
322
323 // Increment the tissue senescence index
324 dstem_senescence_index++;
325 }
326
327 // Calculate root senescence
328 if (TTc >= seneRoot) {
329 // Look back in time to find out how much the tissue grew in the past.
330 double change = assim_rate_root_vec.at(root_senescence_index);
331
332 // Subtract the new growth from the tissue derivative
333 dRoot -= change;
334
335 // Send the lost tissue to the litter
336 dRootLitter += change;
337
338 // Increment the tissue senescence index
339 droot_senescence_index++;
340 }
341
342 if (kRhizome < 0) {
343 // Increment the rhizome senescence index if it is acting as a source
344 drhizome_senescence_index++;
345 }
346
347 // Calculate rhizome senescence
348 if (TTc >= seneRhizome) {
349 // Look back in time to find out how much the tissue grew in the past.
350 double change = assim_rate_rhizome_vec.at(rhizome_senescence_index);
351
352 // Subtract the new growth from the tissue derivative
353 dRhizome -= change;
354
355 // Send the lost tissue to the litter
356 dRhizomeLitter += change;
357
358 // Increment the tissue senescence index
359 drhizome_senescence_index++;
360 }
361
362 // Update the output quantity list
363 update(leafdeathrate_op, dLeafdeathrate);
364
365 update(Leaf_op, dLeaf);
366 update(Stem_op, dStem);
367 update(Root_op, dRoot);
368 update(Rhizome_op, dRhizome);
369 update(Grain_op, dGrain);
370
371 update(LeafLitter_op, dLeafLitter);
372 update(StemLitter_op, dStemLitter);
373 update(RootLitter_op, dRootLitter);
374 update(RhizomeLitter_op, dRhizomeLitter);
375
376 update(stem_senescence_index_op, dstem_senescence_index);
377 update(root_senescence_index_op, droot_senescence_index);
378 update(rhizome_senescence_index_op, drhizome_senescence_index);
379}
