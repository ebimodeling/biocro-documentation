148void partitioning_growth_calculator::do_operation() const
149{
150 double net_assimilation_rate_leaf{0.0};
151 double net_assimilation_rate_stem{0.0};
152 double net_assimilation_rate_root{0.0};
153 double net_assimilation_rate_rhizome{0.0};
154 double net_assimilation_rate_grain{0.0};
155
156 // Calculate the rate of new leaf production
157 if (kLeaf > 0) {
158 net_assimilation_rate_leaf = canopy_assimilation_rate * kLeaf * LeafWS;
159 net_assimilation_rate_leaf = resp(net_assimilation_rate_leaf, mrc1, temp);
160 } else {
161 net_assimilation_rate_leaf = 0.0;
162 }
163
164 // Calculate the rate of new stem production
165 if (kStem >= 0) {
166 net_assimilation_rate_stem = canopy_assimilation_rate * kStem;
167 net_assimilation_rate_stem = resp(net_assimilation_rate_stem, mrc1, temp);
168 } else {
169 net_assimilation_rate_stem = 0.0;
170 }
171
172 // Calculate the rate of new root production
173 if (kRoot > 0) {
174 net_assimilation_rate_root = canopy_assimilation_rate * kRoot;
175 net_assimilation_rate_root = resp(net_assimilation_rate_root, mrc2, temp);
176 } else {
177 net_assimilation_rate_root = 0.0;
178 }
179
180 // Calculate the rate of new rhizome production
181 if (kRhizome > 0) {
182 net_assimilation_rate_rhizome = canopy_assimilation_rate * kRhizome;
183 net_assimilation_rate_rhizome = resp(net_assimilation_rate_rhizome, mrc2, temp);
184 } else {
185 net_assimilation_rate_rhizome = 0.0;
186 }
187
188 // Calculate the rate of grain production
189 if (kGrain > 0 && canopy_assimilation_rate > 0) {
190 net_assimilation_rate_grain = canopy_assimilation_rate * kGrain;
191 } else {
192 net_assimilation_rate_grain = 0.0;
193 }
194
195 // Update the output quantity list
196 update(net_assimilation_rate_leaf_op, net_assimilation_rate_leaf);
197 update(net_assimilation_rate_stem_op, net_assimilation_rate_stem);
198 update(net_assimilation_rate_root_op, net_assimilation_rate_root);
199 update(net_assimilation_rate_rhizome_op, net_assimilation_rate_rhizome);
200 update(net_assimilation_rate_grain_op, net_assimilation_rate_grain);
201}
