129bool depends_on(module_creator* m1, module_creator* m2) {
130 string_vector m1_inputs = get_module_inputs(m1);
131 string_vector m2_outputs = get_module_outputs(m2);
132
133 string_vector result;
134
135 // set_intersection relies on both ranges being sorted, which they are.
136 set_intersection(m1_inputs.begin(), m1_inputs.end(),
137 m2_outputs.begin(), m2_outputs.end(),
138 back_inserter(result));
139
140 // If m1's inputs overlap with m2's outputs (i.e., their intersection is
141 // non-empty), m1 depends on m2:
142 return result.size() != 0;
143}

158Graph get_dependency_graph(mc_vector mcs) {
159
160 int N = mcs.size();
161
162 Graph g(N);
163
164 module_name_map_t name_map = get(vertex_name, g);
165
166 index_map_t index_map = get(vertex_index, g);
167
168
169 vertex_iterator vi;
170 vertex_iterator vi_end;
171 string_vector::size_type i = 0;
172 for (tie(vi, vi_end) = vertices(g); vi != vi_end; ++vi, ++i) {
173 name_map[*vi] = mcs[i];
174
175 // If the adjacency_list VertexList parameter is vecS, then
176 // index_map will be read-only; but it will be set properly so
177 // that the body of this if statement is not executed. On the
178 // other hand, if it is listS, then we must set it manually:
179 if (index_map[*vi] != i) {
180 index_map[*vi] = i;
181 }
182 }
183
184 vertex_iterator vi2;
185 vertex_iterator vi2_end;
186 for (tie(vi, vi_end) = vertices(g); vi != vi_end; ++vi) {
187 for (tie(vi2, vi2_end) = vertices(g); vi2 != vi2_end; ++vi2) {
188 if (depends_on(name_map[*vi], name_map[*vi2])) {
189 add_edge(*vi2, *vi, g);
190 }
191 }
192 }
193
194 return g;
195}
