94void leaf_gbw_nikolov::do_operation() const
95{
96 // This is for about 20 degrees C at 100000 Pa
97 constexpr double volume_of_one_mole_of_air = 24.39e-3; // m^3 / mol
98
99 // Get the temperature difference between the leaf and the air
100 const double delta_t = leaf_temperature - air_temperature; // degrees C
101
102 // Determine the partial pressure of water vapor from the air temperature
103 // and relative humidity
104 const double SWVP = saturation_vapor_pressure(air_temperature); // Pa
105 const double water_vapor_pressure = rh * SWVP; // Pa
106
107 // Calculate the boundary layer conductance. Here we need to convert
108 // stomatal conductance from mol / m^2 / s to m / s.
109 const double gbw = leaf_boundary_layer_conductance_nikolov(
110 windspeed,
111 leafwidth,
112 air_temperature,
113 delta_t,
114 Gs * volume_of_one_mole_of_air,
115 water_vapor_pressure,
116 minimum_gbw * volume_of_one_mole_of_air); // m / s
117
118 // Update the output quantity list
119 update(gbw_op, gbw / volume_of_one_mole_of_air); // mol / m^2 / s
120}
