507string_vector find_euler_requirements(mc_vector mcs)
508{
509 // Get all the module inputs and outputs
510 string_set all_module_inputs =
511 find_unique_module_inputs(std::vector<mc_vector>{mcs});
512
513 string_set all_module_outputs =
514 find_unique_module_outputs(mcs);
515
516 // Make an appropriate state_map that contains them all
517 state_map quantities;
518 for (string_set const& names : std::vector<string_set>{all_module_inputs, all_module_outputs}) {
519 for (std::string const& n : names) {
520 quantities[n] = 0;
521 }
522 }
523
524 // Instantiate each module and check its characterization
525 string_vector euler_requiring_modules;
526 module_vector modules = get_module_vector(mcs, quantities, &quantities);
527 for (size_t i = 0; i < modules.size(); ++i) {
528 if (modules[i]->requires_euler_ode_solver()) {
529 euler_requiring_modules.push_back(mcs[i]->get_name());
530 }
531 }
532
533 return euler_requiring_modules;
534}
