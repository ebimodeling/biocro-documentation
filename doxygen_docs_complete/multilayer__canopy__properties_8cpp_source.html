88 void multilayer_canopy_properties::run() const
89 {
90  // Calculate PAR levels throughout the canopy. Note that sunML specifies
91  // that its `Idir` and `Idiff` arguments should have [micromol / m^2 / s]
92  // units. However, we can leave these quantities as energy flux densities
93  // with [J / m^2 / s] units because the two units are simply related by a
94  // multiplicative conversion factor and the output of sunML is linear with
95  // respect to `Idir` and `Idiff`.
96  struct Light_profile par_profile = sunML(
97  par_incident_direct,
98  par_incident_diffuse,
99  lai,
100  nlayers,
101  cosine_zenith_angle,
102  kd,
103  chil,
104  absorptivity_par,
105  heightf);
106 
107  // Calculate relative humidity levels throughout the canopy
108  double relative_humidity_profile[nlayers];
109  RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile.
110 
111  // Calculate windspeed throughout the canopy
112  double wind_speed_profile[nlayers];
113  WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile.
114 
115  // Calculate leaf nitrogen throughout the canopy
116  double leafN_profile[nlayers];
117  LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile.
118 
119  // Don't calculate anything based on the nitrogen profile
120  if (lnfun != 0) {
121  throw std::logic_error("Thrown by the multilayer_canopy_properties module: lnfun != 0 is not yet supported.");
122  }
123 
124  // Update layer-dependent outputs
125  for (int i = 0; i < nlayers; ++i) {
126  update(sunlit_fraction_ops[i], par_profile.sunlit_fraction[i]);
127  update(shaded_fraction_ops[i], par_profile.shaded_fraction[i]);
128  update(height_ops[i], par_profile.height[i]);
129  update(sunlit_incident_par_ops[i], par_profile.direct_irradiance[i]);
130  update(incident_scattered_par_ops[i], par_profile.scattered_irradiance[i]);
131  update(shaded_incident_par_ops[i], par_profile.diffuse_irradiance[i]);
132  update(incident_average_par_ops[i], par_profile.total_irradiance[i]);
133  update(rh_ops[i], relative_humidity_profile[i]);
134  update(windspeed_ops[i], wind_speed_profile[i]);
135  update(LeafN_ops[i], leafN_profile[i]);
136  }
137 }
