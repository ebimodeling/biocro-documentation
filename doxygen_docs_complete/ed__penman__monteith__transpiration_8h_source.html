90 void ed_penman_monteith_transpiration::do_operation() const
91 {
92  // Determine the total energy available to the leaf
93  const double total_available_energy = solar_energy_absorbed_leaf - long_wave_energy_loss_leaf; // W / m^2
94 
95  // Convert temperatures to Kelvin
96  const double Tak = temperature_air + conversion_constants::celsius_to_kelvin; // Kelvin
97 
98  // Calculate the volume of one mole of a gas at air temperature and pressure
99  // using the ideal gas law
100  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / atmospheric_pressure; // m^3 / mol
101 
102  // Convert conductances to m / s
103  const double gc = conductance_stomatal_h2o * volume_per_mol; // m / s
104  const double ga = conductance_boundary_h2o * volume_per_mol; // m / s
105 
106  // Calculate the transpiration rate
107  // Some notes about units:
108  // 1 W = 1 J / s
109  // 1 J = 1 N * m
110  // 1 N = 1 kg * m / s^2
111  // So 1 J = 1 kg * m^2 / s^2
112  // And 1 W = 1 kg * m^2 / s^3
113  // Units for slope_water_vapor * total_available energy:
114  // kg / m^3 / degree K * W / m^2
115  // = kg / m^3 / degree K * kg * m^2 / s^3 / m^2
116  // = kg^2 / m^3 / s^3 / degree K
117  // Units for latent_heat_vaporization_of_water * psychrometric_parameter * ga * vapor_density_deficit:
118  // J / kg * kg / m^3 / degree K * m / s * kg / m^3
119  // = kg * m^2 / s^2 / kg * kg / m^3 / degree K * m / s * kg / m^3
120  // = kg^2 / m^3 / s^3 / degree K
121  // Units for latent_heat_vaporization_of_water * slope_water_vapor:
122  // J / kg * kg / m^3 / degree K
123  // = kg * m^2 / s^2 / kg * kg / m^3 / degree K
124  // = kg / m / s^2 / degree K
125  // Units for psychrometric_parameter are same as slope_water_vapor
126  // So units for E are kg / m^2 / s
127  const double E_numerator = slope_water_vapor * total_available_energy +
128  latent_heat_vaporization_of_water * psychrometric_parameter * ga * vapor_density_deficit;
129  const double E_denomenator = latent_heat_vaporization_of_water * (slope_water_vapor + psychrometric_parameter * (1 + ga / gc));
130  const double E = E_numerator / E_denomenator; // kg / m^2 / s
131 
132  // Convert E from kg / m^2 / s to mol / m^2 / s using:
133  // 1e3 g / kg
134  // 18 g / mol for water
135  const double transpiration_rate = E * 1.0e3 / 18.0;
136 
137  // Check for error conditions
138  std::map<std::string, bool> errors_to_check = {
139  {"ga cannot be zero", fabs(ga) < calculation_constants::eps_zero}, // divide by zero
140  {"gc cannot be zero", fabs(gc) < calculation_constants::eps_zero}, // divide by zero
141  {"E_denomenator cannot be zero", fabs(E_denomenator) < calculation_constants::eps_zero} // divide by zero
142  };
143 
144  check_error_conditions(errors_to_check, get_name());
145 
146  update(transpiration_rate_op, transpiration_rate);
147 }
