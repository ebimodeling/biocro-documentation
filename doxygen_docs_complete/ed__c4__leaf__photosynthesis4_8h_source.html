179 void ed_c4_leaf_photosynthesis4::get_default(std::vector<double>& guess_vec) const
180 {
181  // Here we try to guess a good starting point based on the availability
182  // of light and water.
183 
184  // When light is scarce, gross CO2 assimilation will be close to zero
185  // and net assimilation will be dominated by the respiration term. Here we
186  // calculate it as in the Collatz model, assuming the leaf is at air temperature.
187  double const temperature_factor = pow(collatz_q10, (temperature_air - 25.0) / 10.0); // dimensionless
188  double const respiration_base = collatz_rd * temperature_factor; // mol / m^2 / s
189  double const respiration_high_temp_inhibition = 1 + exp(0.3 * (temperature_air - 55.0)); // dimensionless
190  double const respiration = respiration_base / respiration_high_temp_inhibition; // mol / m^2 / s
191 
192  // When light is plentiful, gross CO2 assimilation will be limited by CO2 or
193  // Rubisco. Here we calculate these terms as in the Collatz model, assuming
194  // the leaf is at air temperature and the CO2 concentration in the leaf is at the
195  // atmospheric level.
196  double const co2_rate_constant = collatz_k * temperature_factor; // mol / m^2 / s
197  double const assimilation_carbon_limited = co2_rate_constant * mole_fraction_co2_atmosphere; // mol / m^2 / s
198 
199  const double rubisco_capacity_base = collatz_vmax * temperature_factor;
200  const double rubisco_low_temp_inhibition = 1 + exp(0.3 * (collatz_rubisco_temperature_lower - temperature_air));
201  const double rubisco_high_temp_inhibition = 1 + exp(0.3 * (temperature_air - collatz_rubisco_temperature_upper));
202  const double assimilation_rubisco_limited = rubisco_capacity_base / (rubisco_low_temp_inhibition * rubisco_high_temp_inhibition);
203 
204  // Estimate the gross assimilation using a rectangular hyperbola where the independent
205  // variable is the PAR flux. The maximum value should be the carbon limited rate. We have
206  // somewhat arbitrarily decided that the rate should be 60% of the maximum when the PAR
207  // flux is 2000 micromol / m^2 / s. Essentially, we are attempting to replicate a typical
208  // light response curve.
209  double const reference_PAR = 2000 * 1e-6; // mol / m^2 / s
210  double const reference_relative_rate = 0.6; // dimensionless
211  double const PAR_at_half_max = reference_PAR / reference_relative_rate - reference_PAR; // mol / m^2 / s
212  double const assimilation_gross_guess = std::min(assimilation_carbon_limited, assimilation_rubisco_limited) *
213  collatz_PAR_flux /
214  (collatz_PAR_flux + PAR_at_half_max); // mol / m^2 / s
215 
216  // Get a guess for net assimilation by subtracting respiration from the guess for gross
217  // assimilation. Take water stress into account as a multiplicative factor.
218  double const assimilation_net_guess = StomataWS * (assimilation_gross_guess - respiration); // mol / m^2 / s
219 
220  // Make a guess for the Ball-Berry conductance using the guess for net assimilation determined
221  // above. Here we assume the leaf is at air temperature and the concentrations of H2O and CO2 at
222  // the leaf surface are at their atmospheric values.
223  double const ball_berry_index = std::max(assimilation_net_guess, 0.0) * relative_humidity_atmosphere / mole_fraction_co2_atmosphere;
224  double const ball_berry_conductance = ball_berry_slope * ball_berry_index + ball_berry_intercept;
225 
226  // Adjust the conductance according to water availability, treating the Ball-Berry value as a
227  // maximum that occurs when water is plentiful.
228  double const conductance_stomatal_h2o_guess = conductance_stomatal_h2o_min +
229  StomataWS * (ball_berry_conductance - conductance_stomatal_h2o_min);
230 
231  // The guess for leaf temperature is less critical. Just set it equal to ambient temperature.
232  double const temperature_leaf_guess = temperature_air;
233 
234  // Now update the guess vector
235  guess_vec[0] = assimilation_net_guess;
236  guess_vec[1] = conductance_stomatal_h2o_guess;
237  guess_vec[2] = temperature_leaf_guess;
238 }

240 std::vector<std::vector<double>> ed_c4_leaf_photosynthesis4::get_initial_guesses() const
241 {
242  // Get the default guess to use as a starting point
243  std::vector<double> default_guess(3);
244  get_default(default_guess);
245 
246  // Extract the values
247  double const assimilation_net_guess = default_guess[0];
248  double const conductance_stomatal_h2o_guess = default_guess[1];
249  double const temperature_leaf_guess = default_guess[2];
250 
251  std::string message = "Initial guess calculated by ed_c4_leaf_photosynthesis4:\n";
252  char buff[128];
253  sprintf(buff, " assimilation_net_guess = %e\n", assimilation_net_guess);
254  message += std::string(buff);
255  sprintf(buff, " conductance_stomatal_h2o_guess = %e\n", conductance_stomatal_h2o_guess);
256  message += std::string(buff);
257  sprintf(buff, " temperature_leaf_guess = %e\n", temperature_leaf_guess);
258  message += std::string(buff);
259 
260  string_vector names = {
261  "assimilation_net",
262  "conductance_stomatal_h2o",
263  "temperature_leaf"};
264 
265  message += std::string("\nOrder of unknown variable names as supplied:\n");
266  message += std::string(" ") + names[0] + std::string("\n");
267  message += std::string(" ") + names[1] + std::string("\n");
268  message += std::string(" ") + names[2] + std::string("\n");
269 
270  std::sort(names.begin(), names.end());
271 
272  message += std::string("\nOrder of unknown variable names after applying std::sort:\n");
273  message += std::string(" ") + names[0] + std::string("\n");
274  message += std::string(" ") + names[1] + std::string("\n");
275  message += std::string(" ") + names[2] + std::string("\n");
276 
277  // We also want to try a few perturbations surrounding this guess.
278  // For net assimilation, we'll try using a percentage of the guess value as an additive perturbation, but also enforce a minumum.
279  // For stomatal conductance, we'll try using a multiplicative factor for the perturbation
280  // For leaf temperature, we'll just use a fixed step
281  double const perturbation_fraction_an = 0.1; // dimensionless
282  double const min_an_perturbation = 1e-9; // mol / m^2 / s
283  double const an_perturbation = std::max(perturbation_fraction_an * fabs(assimilation_net_guess), min_an_perturbation); // mol / m^2 / s
284  double const perturbation_fraction_gs = 0.1; // dimensionless
285  double const tl_perturbation = 5.0; // degrees C
286 
287  // For An and Gs, we'll add and subtract the perturbation from each guess `num_pts`
288  // times, i.e., we'll try 2 * num_pts + 1 guesses for each parameter. For Tl, we'll
289  // fix num_pts to be 1. So in total, we will try 3 * (2 * num_pts + 1)^2 guesses.
290  int const num_pts = 2;
291  int const num_pts_tl = 1;
292 
293  std::vector<std::vector<double>> input_guesses;
294 
295  message += "\nGuesses from `ed_c4_leaf_photosynthesis4`:\n";
296 
297  for (int i = -num_pts; i <= num_pts; ++i) {
298  double an_value = assimilation_net_guess + an_perturbation * i;
299  for (int j = -num_pts; j <= num_pts; ++j) {
300  double gs_value = conductance_stomatal_h2o_guess * pow(perturbation_fraction_gs, j);
301  for (int k = -num_pts_tl; k <= num_pts_tl; ++k) {
302  double tl_value = temperature_leaf_guess + tl_perturbation * k;
303  input_guesses.push_back({an_value, gs_value, tl_value});
304  sprintf(buff, " an = %e, gs = %e, tl = %e\n", an_value, gs_value, tl_value);
305  message += std::string(buff);
306  }
307  }
308  }
309 
310  message += "\n";
311  if (eclp4_print) {
312  Rprintf(message.c_str());
313  }
314 
315  return input_guesses;
316 }
