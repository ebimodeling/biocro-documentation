188 void multilayer_canopy_properties::run() const
189 {
190  // Calculate PAR levels throughout the canopy. Note that sunML specifies that its
191  // `Idir` and `Idiff` arguments should have [micromol / m^2 / s] units. However,
192  // we can leave these quantities as energy flux densities with [J / m^2 / s] units
193  // because the two units are simply related by a multiplicative conversion factor
194  // and the output of sunML is linear with respect to `Idir` and `Idiff`.
195  struct Light_profile par_profile = sunML(par_incident_direct, par_incident_diffuse,
196  lai, nlayers, cosine_zenith_angle, kd, chil, absorptivity_par, heightf);
197 
198  // Calculate relative humidity levels throughout the canopy
199  double relative_humidity_profile[nlayers];
200  RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile.
201 
202  // Calculate windspeed throughout the canopy
203  double wind_speed_profile[nlayers];
204  WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile.
205 
206  // Calculate leaf nitrogen throughout the canopy
207  double leafN_profile[nlayers];
208  LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile.
209 
210  // Don't calculate anything based on the nitrogen profile
211  if (lnfun != 0) {
212  throw std::logic_error("Thrown by the multilayer_canopy_properties module: lnfun != 0 is not yet supported.");
213  }
214 
215  // Update layer-dependent outputs
216  for (int i = 0; i < nlayers; ++i) {
217  update(sunlit_fraction_ops[i], par_profile.sunlit_fraction[i]);
218  update(shaded_fraction_ops[i], par_profile.shaded_fraction[i]);
219  update(height_ops[i], par_profile.height[i]);
220  update(sunlit_incident_par_ops[i], par_profile.direct_irradiance[i]);
221  update(incident_scattered_par_ops[i], par_profile.scattered_irradiance[i]);
222  update(shaded_incident_par_ops[i], par_profile.diffuse_irradiance[i]);
223  update(incident_average_par_ops[i], par_profile.total_irradiance[i]);
224  update(rh_ops[i], relative_humidity_profile[i]);
225  update(windspeed_ops[i], wind_speed_profile[i]);
226  update(LeafN_ops[i], leafN_profile[i]);
227  }
228 }
