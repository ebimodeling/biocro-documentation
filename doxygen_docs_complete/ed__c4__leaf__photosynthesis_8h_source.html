171  // Get a crappy estimate for an assimilation rate using
172  // a simplified version of the collatz model.
173  //
174  // Here we make the following assumptions, which are
175  // unlikely to be true but may nevertheless help choose
176  // a good starting guess:
177  // (1) the leaf is at 25 C
178  // (i.e., we don't apply any temperature factors)
179  // (2) the CO2 concentration inside the leaf is equal to the atmospheric value
180  // (i.e., Ci = Ca)
181  // (3) gross assimilation is simply the smallest of the CO2, rubisco, and light limited rates
182  // (i.e., no quadratic mixing)
183  const double assimilation_carbon_limited = collatz_k * mole_fraction_co2_atmosphere; // mol / m^2 / s
184  const double assimilation_rubisco_limited = collatz_vmax; // mol / m^2 / s
185  const double assimilation_light_limited = collatz_alpha * collatz_PAR_flux; // mol / m^2 / s
186  const double assimilation_gross = std::min(assimilation_carbon_limited, std::min(assimilation_rubisco_limited, assimilation_light_limited)); // mol / m^2 / s
187  const double assimilation_net_estimate = assimilation_gross - collatz_rd; // mol / m^2 / s
188 
189  // Get a crappy estimation for stomatal conductance using the
190  // Ball-Berry model.
191  //
192  // Here we make the following assumptions, which are
193  // unlikely to be true but may nevertheless help choose
194  // a good starting guess:
195  // (1) the CO2 concentration at the leaf surface is equal to the atmospheric value
196  // (i.e., Cs = Ca)
197  // (2) the relative humidity at the leaf surface is equal to the atmospheric value
198  // (i.e., RHs = RHa)
199  // (3) net assimilation is given by the estimate calculated above
200  const double ball_berry_index = std::max(0.0, assimilation_net_estimate) * relative_humidity_atmosphere / mole_fraction_co2_atmosphere; // mol / m^2 / s
201  const double ball_berry_conductance = ball_berry_intercept + ball_berry_slope * ball_berry_index; // mol / m^2 / s
202  const double conductance_stomatal_h2o_estimate = conductance_stomatal_h2o_min + StomataWS * (ball_berry_conductance - conductance_stomatal_h2o_min); // mol / m^2 / s
203 
204  // Get a crappy estimate for leaf temperature using the Penman-Monteith model.
205  //
206  // Here we make the following assumptions, which are
207  // unlikely to be true but may nevertheless help choose
208  // a good starting guess:
209  // (1) long wave energy loss from the leaf to its environment is zero
210  // (2) boundary layer conductance is 1.2 mol / m^2 / s, an arbitrary value that
211  // used to be hard-coded into the Ball-Berry model
212  const double latent_heat_vaporization_of_water = TempToLHV(temperature_air); // J / kg
213  const double slope_water_vapor = TempToSFS(temperature_air); // kg / m^3 / K
214  const double saturation_water_vapor_pressure = saturation_vapor_pressure(temperature_air); // Pa
215  const double density_of_dry_air = TempToDdryA(temperature_air); // kg / m^3
216  const double saturation_water_vapor_content = saturation_water_vapor_pressure /
217  physical_constants::ideal_gas_constant /
218  (temperature_air + conversion_constants::celsius_to_kelvin) *
219  physical_constants::molar_mass_of_water; // kg / m^3
220  const double vapor_density_deficit = saturation_water_vapor_content * (1 - relative_humidity_atmosphere); // Pa
221  const double psychrometric_parameter = density_of_dry_air *
222  specific_heat_of_air /
223  latent_heat_vaporization_of_water; // kg / m^3 / K
224  const double Tak = temperature_air + conversion_constants::celsius_to_kelvin; // Kelvin
225  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / atmospheric_pressure; // m^3 / mol
226 
227  const double total_available_energy = solar_energy_absorbed_leaf; // J / m^2 / s
228  const double gc = conductance_stomatal_h2o_estimate * volume_per_mol; // m / s
229  const double ga = 1.2 * volume_per_mol; // m / s
230  const double delta_t_numerator = total_available_energy * (1 / ga + 1 / gc) - latent_heat_vaporization_of_water * vapor_density_deficit;
231  const double delta_t_denomenator = latent_heat_vaporization_of_water * (slope_water_vapor + psychrometric_parameter * (1 + ga / gc));
232  const double delta_t = delta_t_numerator / delta_t_denomenator; // deg. C
233  const double temperature_leaf_estimate = temperature_air + delta_t; // deg. C
234 
235  // Get a crappy estimate for free boundary layer conductance using the Nikolov model.
236  //
237  // Here we make the following assumptions, which are
238  // unlikely to be true but may nevertheless help choose
239  // a good starting guess:
240  // (1) the leaf temperature is the value estimated above
241  // (2) the relative humidity at the leaf surface is equal to the atmospheric value
242  const double Tlk = temperature_leaf_estimate + conversion_constants::celsius_to_kelvin; // Kelvin
243  const double mole_fraction_h2o_atmosphere = relative_humidity_atmosphere * saturation_water_vapor_pressure / atmospheric_pressure; // dimensionless
244  const double mole_fraction_h2o_leaf_surface = mole_fraction_h2o_atmosphere;
245  const double Tvdiff = Tlk / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_leaf_surface) -
246  Tak / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_atmosphere); // deg. C or K
247  const double gbv_free_estimate = nikolov_ce *
248  pow(Tlk, nikolov::temperature_exponent) *
249  pow((Tlk + nikolov::temperature_offset) / atmospheric_pressure, 0.5) *
250  pow(fabs(Tvdiff) / leafwidth, 0.25) /
251  volume_per_mol; // mol / m^2 / s
