140 void soybean_development_rate_calculator::do_operation() const
141 {
142  double soybean_development_rate; // day^-1
143 
144 
145  if (DVI < -1) {
146  // error, DVI out of bounds, this should never occur unless initial DVI
147  // state is less than -1.
148  soybean_development_rate = 0; // day^-1
149 
150  } else if (DVI < 0) {
151  // 1. Sowing to emergence
152 
153  double temp_diff = temp - Tbase_emr; // degrees C
154  soybean_development_rate = temp_diff / TTemr_threshold; // day^-1
155 
156  } else if (DVI < 0.333) {
157  // 2a. Emergence - V0 (cotyledon stage); r = Rmax * f(T)
158 
159  soybean_development_rate = Rmax_emrV0 * tempFunc(temp, Tmin_emrV0,
160  Topt_emrV0, Tmax_emrV0) / 3.0; // day^-1
161  } else if (DVI < 0.667) {
162  // 2b. V0 (cotyledon) - R0 (end of floral induction); r = Rmax * f(P)
163 
164  const double Rmax_V0R0 = 0.0294 + 0.1561 / maturity_group; // day^-1;
165  // Setiyono et al., 2007, Fig. 12a
166 
167  const double Popt_V0R0 = 12.759 - 0.388 * maturity_group -
168  0.058 * pow(maturity_group,2); // hours; Setiyono et al., 2007, Fig 4
169  const double Pcrit_V0R0 = 27.275 - 0.493 * maturity_group -
170  0.066 * pow(maturity_group,2); // hours; Setiyono et al., 2007, Fig 4
171 
172  double day_length_civil = 1.072 * day_length; // hours;
173  // Converts day length calculated based on solar elevation = -0.83 degrees
174  // to civil day length (based on solar elevation angle = -6 degrees)
175 
176  soybean_development_rate = Rmax_V0R0 *
177  photoFunc(day_length_civil, Popt_V0R0, Pcrit_V0R0) / 3.0; // day^-1
178 
179  } else if (DVI < 1) {
180  // 2c. R0 (end of floral induction) - R1 (flowering); r = Rmax * f(T)
181 
182  const double Rmax_R0R1 = (0.2551 + 0.0965 * maturity_group) /
183  (1 + 2.1107 * maturity_group); // day^-1; Setiyono et al., 2007, Fig 11a
184 
185  soybean_development_rate = Rmax_R0R1 *
186  tempFunc(temp, Tmin_R0R1, Topt_R0R1, Tmax_R0R1) / 3.0; // day^-1
187 
188  } else {
189  // 3. Reproductive stages, R1 (flowering) - R7 (maturity);
190  // r = Rmax * f(T) * f(P)
191 
192  const double Rmax_R1R7 = (0.0563 + 0.0228 * maturity_group) /
193  (1 + 1.9683 * maturity_group); // day^-1; Setiyono et al., 2007, Fig 11d
194 
195  const double Popt_R1R7 = 10.6595 + 2.9706 / maturity_group; // hrs;
196  // Setiyono et al., 2007, Fig 12c
197 
198  const double Pcrit_R1R7 = 16.1257 + 4.3143 / maturity_group; // hrs;
199  // Setiyono et al., 2007, Fig 12c
200 
201  soybean_development_rate = Rmax_R1R7 *
202  tempFunc(temp, Tmin_R1R7, Topt_R1R7, Tmax_R1R7) *
203  photoFunc(day_length, Popt_R1R7, Pcrit_R1R7); // day^-1
204  }
205 
206  double development_rate_per_hour = soybean_development_rate / 24.0; // hr^-1
207 
208  // Update the output parameter list
209  update(development_rate_per_hour_op, development_rate_per_hour);
210 }

212 double tempFunc(double T, double Tmin, double Topt, double Tmax)
213 {
214  double fT; // dimensionless
215 
216  if (T > Tmin && T < Tmax) {
217  double alpha = log(2.0) / log((Tmax - Tmin) / (Topt - Tmin)); // dimensionless
218  double fT_num_pt1 = 2.0 * pow(T - Tmin, alpha) * pow(Topt - Tmin, alpha); // dimensionless
219  double fT_num_pt2 = pow(T - Tmin, 2.0 * alpha); // dimensionless
220  double fT_denom = pow(Topt - Tmin, 2.0 * alpha); // dimensionless
221 
222  fT = (fT_num_pt1 - fT_num_pt2) / fT_denom; // dimensionless
223  } else {
224  fT = 0.0; // dimensionless
225  }
226 
227  return fT; // dimensionless
228 }

230 double photoFunc(double P, double Popt, double Pcrit)
231 {
232  double m = 3.0; // hrs; Setiyono et al., 2007 Equation 4, value in text below equation
233  double fP; // dimensionless
234 
235  if (P <= Popt) {
236  fP = 1.0; // dimensionless
237 
238  } else if (P < Pcrit) {
239 
240  double alpha = log(2.0) / log(1.0 + (Pcrit - Popt) / m); // dimensionless
241  double fP_pt1 = 1.0 + (P - Popt) / m; // dimensionless
242  double fP_pt2 = (Pcrit - P) / (Pcrit - Popt); // dimensionless
243  double fP_pt3 = (Pcrit - Popt) / m; // dimensionless
244 
245  fP = pow(fP_pt1 * pow(fP_pt2, fP_pt3), alpha); // dimensionless
246 
247  } else {
248  fP = 0.0; // dimensionless
249  }
250 
251  return fP; // dimensionless
252 }
