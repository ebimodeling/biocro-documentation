96 template <typename equation_ptr_type, typename x_vector_type, typename time_type, typename f_vector_type, typename matrix_type>
97 void calculate_jacobian(
98  equation_ptr_type const& equation_ptr,
99  x_vector_type const& x,
100  time_type t,
101  f_vector_type const& f_current,
102  matrix_type& jacobi)
103 {
104  size_t n = x.size();
105 
106  // Make a vector to store the perturbed f(x,t)
107  f_vector_type f_perturbed(n);
108 
109  // Perturb each element x_i of the input vector to find df_j(x,t)/dx_i,
110  // which is stored at jacobi(j,i)
111  double h;
112  x_vector_type x_perturbed = x;
113 
114  for (size_t i = 0; i < n; i++) {
115  // Detemine the step size h by taking a fraction of x[i]: h = x[i] * eps_deriv.
116  // Ensure that the step size h is close to this value but is exactly representable
117  // (see Numerical Recipes in C, 2nd ed., Section 5.7)
118  h = x[i] * calculation_constants::eps_deriv;
119  if (h == 0.0) {
120  h = calculation_constants::eps_deriv * calculation_constants::eps_deriv;
121  }
122  double temp = x[i] + h;
123  h = temp - x[i];
124 
125  // Calculate the new function value
126  x_perturbed[i] = x[i] + h; // Add h to the ith state variable
127  evaluate_equations(equation_ptr, x_perturbed, t, f_perturbed); // Calculate f_perturbed
128 
129  // Store the results in the Jacobian matrix
130  for (size_t j = 0; j < n; j++) {
131  jacobi(j, i) = (f_perturbed[j] - f_current[j]) / h;
132  }
133 
134  // Reset the ith state variable
135  x_perturbed[i] = x[i];
136  }
137 }

218  // Use a forward step whenever possible
219  // Detemine the step size h by taking a fraction of t: h = t * eps_deriv.
