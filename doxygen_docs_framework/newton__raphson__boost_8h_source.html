18 boost::numeric::ublas::vector<double> get_newton_raphson_step_boost(
19  boost::numeric::ublas::vector<double> F_x_0,
20  boost::numeric::ublas::matrix<double> jacobian)
21 {
22  // Determine the inverse of the Jacobian matrix: J^(-1)
23  // See e.g. http://www.crystalclearsoftware.com/cgi-bin/boost_wiki/wiki.pl?LU_Matrix_Inversion
24  boost::numeric::ublas::permutation_matrix<std::size_t> pm(jacobian.size1()); // create a permutation matrix for the LU-factorization
25  int res = boost::numeric::ublas::lu_factorize(jacobian, pm); // perform the LU-factorization
26  if (res != 0) {
27  throw std::runtime_error(std::string("Thrown by newton_raphson_boost: LU factorization of Jacobian matrix failed!"));
28  }
29  boost::numeric::ublas::matrix<double> jacobian_inverse =
30  boost::numeric::ublas::identity_matrix<double>(jacobian.size1()); // initialize the inverse to be an identity matrix
31  boost::numeric::ublas::lu_substitute(jacobian, pm, jacobian_inverse); // find the inverse by back-substitution
32 
33  // Calculate the change in x according to the Newton-Raphson formula: dx = -J^(-1) * F
34  return -1.0 * boost::numeric::ublas::prod(jacobian_inverse, F_x_0);
35 }
