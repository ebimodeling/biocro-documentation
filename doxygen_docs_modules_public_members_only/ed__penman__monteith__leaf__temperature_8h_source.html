98 void ed_penman_monteith_leaf_temperature::do_operation() const
99 {
100  // Determine the total energy available to the leaf
101  const double total_available_energy = *solar_energy_absorbed_leaf_ip - *long_wave_energy_loss_leaf_ip; // W / m^2
102 
103  // Convert temperatures to Kelvin
104  const double Tak = *temperature_air_ip + conversion_constants::celsius_to_kelvin; // Kelvin
105 
106  // Calculate the volume of one mole of a gas at air temperature and pressure
107  // using the ideal gas law
108  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
109 
110  // Convert conductances to m / s
111  const double gc = *conductance_stomatal_h2o_ip * volume_per_mol; // m / s
112  const double ga = *conductance_boundary_h2o_ip * volume_per_mol; // m / s
113 
114  const double delta_t_numerator = total_available_energy * (1 / ga + 1 / gc) - *latent_heat_vaporization_of_water_ip * *vapor_density_deficit_ip;
115  const double delta_t_denomenator = *latent_heat_vaporization_of_water_ip * (*slope_water_vapor_ip + *psychrometric_parameter_ip * (1 + ga / gc));
116  const double delta_t = delta_t_numerator / delta_t_denomenator;
117 
118  const double temperature_leaf = *temperature_air_ip + delta_t;
119 
120  // Check for error conditions
121  std::map<std::string, bool> errors_to_check = {
122  {"ga cannot be zero", fabs(ga) < calculation_constants::eps_zero}, // divide by zero
123  {"gc cannot be zero", fabs(gc) < calculation_constants::eps_zero}, // divide by zero
124  {"delta_t_denomenator cannot be zero", fabs(delta_t_denomenator) < calculation_constants::eps_zero} // divide by zero
125  };
126 
127  check_error_conditions(errors_to_check, get_name());
128 
129  update(temperature_leaf_op, temperature_leaf);
130 }

228 std::vector<std::vector<double>> ed_p_m_temperature_solve::get_initial_guesses() const
229 {
230  // We want to avoid a spurious high temperature root that may be present when
231  // windspeed is low, so we try a low temperature first. If that fails, it is
232  // likely because the cusp is lurking at low temperatures. In that case, a
233  // higher starting temperature should work.
234  double const temperature_offset = 10.0; // degrees C
235  return std::vector<std::vector<double>>{
236  {temperature_air - temperature_offset},
237  {temperature_air + temperature_offset}};
238 }
