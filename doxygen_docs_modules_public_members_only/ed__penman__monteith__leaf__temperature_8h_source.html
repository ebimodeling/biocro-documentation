86 void ed_penman_monteith_leaf_temperature::do_operation() const
87 {
88  // Determine the total energy available to the leaf
89  const double total_available_energy = *solar_energy_absorbed_leaf_ip - *long_wave_energy_loss_leaf_ip; // W / m^2
90 
91  // Convert temperatures to Kelvin
92  const double Tak = *temperature_air_ip + conversion_constants::celsius_to_kelvin; // Kelvin
93 
94  // Calculate the volume of one mole of a gas at air temperature and pressure
95  // using the ideal gas law
96  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
97 
98  // Convert conductances to m / s
99  const double gc = *conductance_stomatal_h2o_ip * volume_per_mol; // m / s
100  const double ga = *conductance_boundary_h2o_ip * volume_per_mol; // m / s
101 
102  const double delta_t_numerator = total_available_energy * (1 / ga + 1 / gc) - *latent_heat_vaporization_of_water_ip * *vapor_density_deficit_ip;
103  const double delta_t_denomenator = *latent_heat_vaporization_of_water_ip * (*slope_water_vapor_ip + *psychrometric_parameter_ip * (1 + ga / gc));
104  const double delta_t = delta_t_numerator / delta_t_denomenator;
105 
106  const double temperature_leaf = *temperature_air_ip + delta_t;
107 
108  // Check for error conditions
109  std::map<std::string, bool> errors_to_check = {
110  {"ga cannot be zero", fabs(ga) < calculation_constants::eps_zero}, // divide by zero
111  {"gc cannot be zero", fabs(gc) < calculation_constants::eps_zero}, // divide by zero
112  {"delta_t_denomenator cannot be zero", fabs(delta_t_denomenator) < calculation_constants::eps_zero} // divide by zero
113  };
114 
115  check_error_conditions(errors_to_check, get_name());
116 
117  update(temperature_leaf_op, temperature_leaf);
118 }

207 std::vector<std::vector<double>> ed_p_m_temperature_solve::get_initial_guesses() const
208 {
209  // We want to avoid a spurious high temperature root that may be present when
210  // windspeed is low, so we try a low temperature first. If that fails, it is
211  // likely because the cusp is lurking at low temperatures. In that case, a
212  // higher starting temperature should work.
213  double const temperature_offset = 10.0; // degrees C
214  return std::vector<std::vector<double>>{
215  {temperature_air - temperature_offset},
216  {temperature_air + temperature_offset}};
217 }
