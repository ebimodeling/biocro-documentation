135 void ed_nikolov_conductance_forced::do_operation() const
136 {
137  // Convert temperatures to Kelvin
138  const double Tak = *temperature_air_ip + conversion_constants::celsius_to_kelvin; // Kelvin
139 
140  // Calculate the volume of one mole of a gas at air temperature and pressure
141  // using the ideal gas law
142  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
143 
144  // Calculate the forced boundary layer conductance for water vapor
145  // using Equation 29 in Nikolov et al. Note that we have used
146  // volume_per_mol to convert from m / s to mol / m^2 / s (see
147  // class description above for more details).
148  const double gbv_forced = *nikolov_cf_ip *
149  pow(Tak, nikolov::temperature_exponent) *
150  pow((Tak + nikolov::temperature_offset) * *windspeed_ip / (*leafwidth_ip * *atmospheric_pressure_ip), 0.5) /
151  volume_per_mol; // mol / m^2 / s
152 
153  // Check for error conditions
154  std::map<std::string, bool> errors_to_check = {
155  {"atmospheric_pressure cannot be zero", fabs(*atmospheric_pressure_ip) < calculation_constants::eps_zero}, // divide by zero
156  {"volume_per_mol cannot be zero", fabs(volume_per_mol) < calculation_constants::eps_zero}, // divide by zero
157  {"leaf_width cannot be zero", fabs(*leafwidth_ip) < calculation_constants::eps_zero} // divide by zero
158  };
159 
160  check_error_conditions(errors_to_check, get_name());
161 
162  // Update the output quantity list
163  update(conductance_boundary_h2o_forced_op, gbv_forced);
164 }

246 void ed_nikolov_conductance_free::do_operation() const
247 {
248  // Convert temperatures to Kelvin
249  const double Tak = *temperature_air_ip + conversion_constants::celsius_to_kelvin; // Kelvin
250  const double Tlk = *temperature_leaf_ip + conversion_constants::celsius_to_kelvin; // Kelvin
251 
252  // Calculate the volume of one mole of a gas at air temperature and pressure
253  // using the ideal gas law
254  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
255 
256  // Calculate the water vapor mole fraction at the leaf surface that would occur
257  // if boundary_layer_conductance = free_boundary_layer_conductance. This calculation
258  // is analogous to Equation 35 in Nikolov et al., which was originally derived by
259  // requiring equal energy fluxes in the boundary layer and across the stomata. Here
260  // we use molecular fluxes instead. Also note that our concentration at the leaf
261  // surface is equivalent to Nikolov's concentration in the boundary layer.
262  const double mole_fraction_h2o_intercellular = saturation_vapor_pressure(*temperature_leaf_ip) / *atmospheric_pressure_ip; // dimensionless
263  const double h2o_mfs_numer = *conductance_boundary_h2o_free_ip * *mole_fraction_h2o_atmosphere_ip +
264  *conductance_stomatal_h2o_ip * mole_fraction_h2o_intercellular;
265  const double h2o_mfs_denom = *conductance_boundary_h2o_free_ip + *conductance_stomatal_h2o_ip;
266  const double mole_fraction_h2o_leaf_surface = h2o_mfs_numer / h2o_mfs_denom; // dimensionless
267 
268  // Calculate the virtual temperature difference using Equation 34 of Nikolov et al.
269  // Note that the original equation contains terms of partial_pressure/atmospheric_pressure;
270  // We have replaced those terms with equivalent mole fractions.
271  const double Tvdiff = Tlk / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_leaf_surface) -
272  Tak / (1.0 - nikolov::Tvdiff_factor * *mole_fraction_h2o_atmosphere_ip); // deg. C or K
273 
274  // Calculate the free boundary layer conductance using Equation 33 from Nikolov et al.
275  // Note that we have used volume_per_mol to convert from m / s to mol / m^2 / s (see
276  // class description above for more details).
277  const double gbv_free = *nikolov_ce_ip *
278  pow(Tlk, nikolov::temperature_exponent) *
279  pow((Tlk + nikolov::temperature_offset) / *atmospheric_pressure_ip, 0.5) *
280  pow(fabs(Tvdiff) / *leafwidth_ip, 0.25) /
281  volume_per_mol; // mol / m^2 / s
282 
283  // Check for error conditions
284  std::map<std::string, bool> errors_to_check = {
285  {"atmospheric_pressure cannot be zero", fabs(*atmospheric_pressure_ip) < calculation_constants::eps_zero}, // divide by zero
286  {"volume_per_mol cannot be zero", fabs(volume_per_mol) < calculation_constants::eps_zero}, // divide by zero
287  {"leaf_width cannot be zero", fabs(*leafwidth_ip) < calculation_constants::eps_zero}, // divide by zero
288  {"the sum of water vapor conductances cannot be zero", fabs(h2o_mfs_denom) < calculation_constants::eps_zero}, // divide by zero
289  {"leaf temperature cannot be below absolute zero", Tlk < calculation_constants::eps_zero} // pow would have imaginary output
290  };
291 
292  check_error_conditions(errors_to_check, get_name());
293 
294  // Update the output quantity list
295  update(conductance_boundary_h2o_free_op, gbv_free);
296  update(nikolov_virtual_temperature_difference_op, Tvdiff);
297 }
