152  latent_heat_vaporization_of_water; // J / m^2 / s
153 
154  // Apply energy balance to calculate the sensible heat loss
155  double const sensible_heat_loss =
156  solar_energy_absorbed_leaf - long_wave_energy_loss_leaf - et_energy; // J / m^2 / s
157 
158  // Use the sensible heat loss to calculate the leaf temperature
159  double const molar_mass_air = 28.97e-3; // kg / mol
160  double const delta_t = sensible_heat_loss /
161  specific_heat_of_air /

163  conductance_boundary_h2o; // degrees C
164  double const temperature_leaf = temperature_air + delta_t; // degrees C
165
