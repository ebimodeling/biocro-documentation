137 void ed_multilayer_canopy_integrator::run() const
138 {
139  const double LAIc = lai / nlayers;
140  double canopy_assimilation_rate = 0;
141  double canopy_transpiration_rate = 0;
142  double canopy_conductance = 0;
143  double GrossAssim = 0;
144 
145  // Integrate assimilation, transpiration, and conductance throughout the
146  // canopy
147  for (int i = 0; i < nlayers; ++i) {
148  const double sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
149  const double shaded_lai = *shaded_fraction_ips[i] * LAIc;
150 
151  canopy_assimilation_rate +=
152  *sunlit_assimilation_net_ips[i] * sunlit_lai +
153  *shaded_assimilation_net_ips[i] * shaded_lai;
154 
155  canopy_transpiration_rate +=
156  *sunlit_transpiration_rate_ips[i] * sunlit_lai +
157  *shaded_transpiration_rate_ips[i] * shaded_lai;
158 
159  canopy_conductance +=
160  *sunlit_conductance_stomatal_h2o_ips[i] * sunlit_lai +
161  *shaded_conductance_stomatal_h2o_ips[i] * shaded_lai;
162 
163  GrossAssim +=
164  *sunlit_assimilation_gross_ips[i] * sunlit_lai +
165  *shaded_assimilation_gross_ips[i] * shaded_lai;
166  }
167 
168  // Modify net assimilation to account for respiration
169  // Note: this was originally only done for the C3 canopy
170  // Note: it seems like this should not be necessary since the assimilation
171  // model includes respiration
172  canopy_assimilation_rate *= (1.0 - growth_respiration_fraction);
173 
174  // For assimilation, we need to convert mol / m^2 / s into Mg / ha / hr,
175  // assuming that all carbon is converted into biomass in the form of glucose
176  // (C6H12O6), i.e., six assimilated CO2 molecules contribute one glucose
177  // molecule. Using the molar mass of glucose in kg / mol, the conversion can
178  // be accomplished with the following factor:
179  // (1 glucose / 6 CO2) * (3600 s / hr) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
180  // = 6e3 s * Mg * m^2 / (hr * kg * ha)
181  const double cf = physical_constants::molar_mass_of_glucose * 6e3; // (Mg / ha / hr) / (mol / m^2 / s)
182 
183  // For transpiration, we need to convert mol / m^2 / s into Mg / ha / hr
184  // using the molar mass of water in kg / mol, which can be accomplished by
185  // the following conversion factor:
186  // (3600 s / hr) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
187  // = 36e3 s * Mg * m^2 / (hr * kg * ha)
188  const double cf2 = physical_constants::molar_mass_of_water * 36e3; // (Mg / ha / hr) / (mol / m^2 / s)
189 
190  // Convert conductance values from mol / m^2 / s to mmol / m^2 / s using 1e3 mmol / mol
191 
192  update(canopy_assimilation_rate_op, canopy_assimilation_rate * cf);
193  update(GrossAssim_op, GrossAssim * cf);
194  update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
195  update(canopy_conductance_op, canopy_conductance * 1e3);
196 }
