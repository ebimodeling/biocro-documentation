89  double Stem = *Stem_ip;
90  double Root = *Root_ip;
91  double Rhizome = *Rhizome_ip;
92 
93  // In the Thornley senescence model, sensescence is proportional to structural mass
94  // e.g. senescence_leaf = kLeaf * Leaf, for 0 < kLeaf < 1
95  // This is quite general. To see why, imagine using Michaelis-Menton kinetics here.
96  // In this case, we would have a catalyzed reaction turning structural carbon into
97  // substrate or litter
98  // The rate of change of the structural carbon concentration would be
99  // proportional to [structural carbon]/(Km + [structural carbon])
100  // And so the rate of change of total structural carbon would be proportional
101  // to volume * [structural carbon]/(Km + [structural carbon])
102  // For a tissue, volume is proportional to the amount of structural carbon
103  // In this case, [structural carbon] is constant
104  // E.g. we have a rate proportional to Leaf * constant/(Km + constant)
105  // We can absorb the constant into the proportionality constant, ending up with
106  // senescence_leaf = kLeaf * Leaf again
107  // So ultimately this is the same as the simpler assumption
108  // Here we do not allow the grain to senesce
109 
110  double senescence_leaf = kLeaf * Leaf;
111  double senescence_stem = kStem * Stem;
112  double senescence_root = kRoot * Root;
113  double senescence_rhizome = kRhizome * Rhizome;
114 
115  // Update the output quantity list
116  update(senescence_leaf_op, senescence_leaf);
117  update(senescence_stem_op, senescence_stem);
118  update(senescence_root_op, senescence_root);
119  update(senescence_rhizome_op, senescence_rhizome);
120 }
