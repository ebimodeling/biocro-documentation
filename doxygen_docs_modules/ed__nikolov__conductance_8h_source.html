129 void ed_nikolov_conductance_forced::do_operation() const
130 {
131  // Convert temperatures to Kelvin
132  const double Tak = *temperature_air_ip + conversion_constants::celsius_to_kelvin; // Kelvin
133 
134  // Calculate the volume of one mole of a gas at air temperature and pressure
135  // using the ideal gas law
136  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
137 
138  // Calculate the forced boundary layer conductance for water vapor
139  // using Equation 29 in Nikolov et al. Note that we have used
140  // volume_per_mol to convert from m / s to mol / m^2 / s (see
141  // class description above for more details).
142  const double gbv_forced = *nikolov_cf_ip *
143  pow(Tak, nikolov::temperature_exponent) *
144  pow((Tak + nikolov::temperature_offset) * *windspeed_ip / (*leafwidth_ip * *atmospheric_pressure_ip), 0.5) /
145  volume_per_mol; // mol / m^2 / s
146 
147  // Check for error conditions
148  std::map<std::string, bool> errors_to_check = {
149  {"atmospheric_pressure cannot be zero", fabs(*atmospheric_pressure_ip) < calculation_constants::eps_zero}, // divide by zero
150  {"volume_per_mol cannot be zero", fabs(volume_per_mol) < calculation_constants::eps_zero}, // divide by zero
151  {"leaf_width cannot be zero", fabs(*leafwidth_ip) < calculation_constants::eps_zero} // divide by zero
152  };
153 
154  check_error_conditions(errors_to_check, get_name());
155 
156  // Update the output quantity list
157  update(conductance_boundary_h2o_forced_op, gbv_forced);
158 }

236 void ed_nikolov_conductance_free::do_operation() const
237 {
238  // Convert temperatures to Kelvin
239  const double Tak = *temperature_air_ip + conversion_constants::celsius_to_kelvin; // Kelvin
240  const double Tlk = *temperature_leaf_ip + conversion_constants::celsius_to_kelvin; // Kelvin
241 
242  // Calculate the volume of one mole of a gas at air temperature and pressure
243  // using the ideal gas law
244  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / *atmospheric_pressure_ip; // m^3 / mol
245 
246  // Calculate the water vapor mole fraction at the leaf surface that would occur
247  // if boundary_layer_conductance = free_boundary_layer_conductance. This calculation
248  // is analogous to Equation 35 in Nikolov et al., which was originally derived by
249  // requiring equal energy fluxes in the boundary layer and across the stomata. Here
250  // we use molecular fluxes instead. Also note that our concentration at the leaf
251  // surface is equivalent to Nikolov's concentration in the boundary layer.
252  const double mole_fraction_h2o_intercellular = saturation_vapor_pressure(*temperature_leaf_ip) / *atmospheric_pressure_ip; // dimensionless
253  const double h2o_mfs_numer = *conductance_boundary_h2o_free_ip * *mole_fraction_h2o_atmosphere_ip +
254  *conductance_stomatal_h2o_ip * mole_fraction_h2o_intercellular;
255  const double h2o_mfs_denom = *conductance_boundary_h2o_free_ip + *conductance_stomatal_h2o_ip;
256  const double mole_fraction_h2o_leaf_surface = h2o_mfs_numer / h2o_mfs_denom; // dimensionless
257 
258  // Calculate the virtual temperature difference using Equation 34 of Nikolov et al.
259  // Note that the original equation contains terms of partial_pressure/atmospheric_pressure;
260  // We have replaced those terms with equivalent mole fractions.
261  const double Tvdiff = Tlk / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_leaf_surface) -
262  Tak / (1.0 - nikolov::Tvdiff_factor * *mole_fraction_h2o_atmosphere_ip); // deg. C or K
263 
264  // Calculate the free boundary layer conductance using Equation 33 from Nikolov et al.
265  // Note that we have used volume_per_mol to convert from m / s to mol / m^2 / s (see
266  // class description above for more details).
267  const double gbv_free = *nikolov_ce_ip *
268  pow(Tlk, nikolov::temperature_exponent) *
269  pow((Tlk + nikolov::temperature_offset) / *atmospheric_pressure_ip, 0.5) *
270  pow(fabs(Tvdiff) / *leafwidth_ip, 0.25) /
271  volume_per_mol; // mol / m^2 / s
272 
273  // Check for error conditions
274  std::map<std::string, bool> errors_to_check = {
275  {"atmospheric_pressure cannot be zero", fabs(*atmospheric_pressure_ip) < calculation_constants::eps_zero}, // divide by zero
276  {"volume_per_mol cannot be zero", fabs(volume_per_mol) < calculation_constants::eps_zero}, // divide by zero
277  {"leaf_width cannot be zero", fabs(*leafwidth_ip) < calculation_constants::eps_zero}, // divide by zero
278  {"the sum of water vapor conductances cannot be zero", fabs(h2o_mfs_denom) < calculation_constants::eps_zero}, // divide by zero
279  {"leaf temperature cannot be below absolute zero", Tlk < calculation_constants::eps_zero} // pow would have imaginary output
280  };
281 
282  check_error_conditions(errors_to_check, get_name());
283 
284  // Update the output quantity list
285  update(conductance_boundary_h2o_free_op, gbv_free);
286  update(nikolov_virtual_temperature_difference_op, Tvdiff);
287 }
