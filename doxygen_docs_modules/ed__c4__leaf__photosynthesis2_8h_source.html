152 void ed_c4_leaf_photosynthesis2::get_default(std::vector<double>& guess_vec) const
153 {
154  // Here we try to guess a good starting point based on the availability
155  // of light and water.
156 
157  // When light is scarce, gross CO2 assimilation will be close to zero
158  // and net assimilation will be dominated by the respiration term. Here we
159  // calculate it as in the Collatz model, assuming the leaf is at air temperature.
160  double const temperature_factor = pow(collatz_q10, (temperature_air - 25.0) / 10.0); // dimensionless
161  double const respiration_base = collatz_rd * temperature_factor; // mol / m^2 / s
162  double const respiration_high_temp_inhibition = 1 + exp(0.3 * (temperature_air - 55.0)); // dimensionless
163  double const respiration = respiration_base / respiration_high_temp_inhibition; // mol / m^2 / s
164 
165  // When light is plentiful, gross CO2 assimilation will be limited by CO2 or
166  // Rubisco. Here we calculate these terms as in the Collatz model, assuming
167  // the leaf is at air temperature and the CO2 concentration in the leaf is at the
168  // atmospheric level.
169  double const co2_rate_constant = collatz_k * temperature_factor; // mol / m^2 / s
170  double const assimilation_carbon_limited = co2_rate_constant * mole_fraction_co2_atmosphere; // mol / m^2 / s
171 
172  const double rubisco_capacity_base = collatz_vmax * temperature_factor;
173  const double rubisco_low_temp_inhibition = 1 + exp(0.3 * (collatz_rubisco_temperature_lower - temperature_air));
174  const double rubisco_high_temp_inhibition = 1 + exp(0.3 * (temperature_air - collatz_rubisco_temperature_upper));
175  const double assimilation_rubisco_limited = rubisco_capacity_base / (rubisco_low_temp_inhibition * rubisco_high_temp_inhibition);
176 
177  // Estimate the gross assimilation using a rectangular hyperbola where the independent
178  // variable is the PAR flux. The maximum value should be the carbon limited rate. We have
179  // somewhat arbitrarily decided that the rate should be 60% of the maximum when the PAR
180  // flux is 2000 micromol / m^2 / s. Essentially, we are attempting to replicate a typical
181  // light response curve.
182  double const reference_PAR = 2000 * 1e-6; // mol / m^2 / s
183  double const reference_relative_rate = 0.6; // dimensionless
184  double const PAR_at_half_max = reference_PAR / reference_relative_rate - reference_PAR; // mol / m^2 / s
185  double const assimilation_gross_guess = std::min(assimilation_carbon_limited, assimilation_rubisco_limited) *
186  collatz_PAR_flux /
187  (collatz_PAR_flux + PAR_at_half_max); // mol / m^2 / s
188 
189  // Get a guess for net assimilation by subtracting respiration from the guess for gross
190  // assimilation. Take water stress into account as a multiplicative factor.
191  double const assimilation_net_guess = StomataWS * (assimilation_gross_guess - respiration); // mol / m^2 / s
192 
193  // Make a guess for the Ball-Berry conductance using the guess for net assimilation determined
194  // above. Here we assume the leaf is at air temperature and the concentrations of H2O and CO2 at
195  // the leaf surface are at their atmospheric values.
196  double const ball_berry_index = std::max(assimilation_net_guess, 0.0) * relative_humidity_atmosphere / mole_fraction_co2_atmosphere;
197  double const ball_berry_conductance = ball_berry_slope * ball_berry_index + ball_berry_intercept;
198 
199  // Adjust the conductance according to water availability, treating the Ball-Berry value as a
200  // maximum that occurs when water is plentiful.
201  double const conductance_stomatal_h2o_guess = conductance_stomatal_h2o_min +
202  StomataWS * (ball_berry_conductance - conductance_stomatal_h2o_min);
203 
204  // The guess for leaf temperature is less critical. Just set it equal to ambient temperature.
205  double const temperature_leaf_guess = temperature_air;
206 
207  // Now update the guess vector
208  guess_vec[0] = assimilation_net_guess;
209  guess_vec[1] = conductance_stomatal_h2o_guess;
210  guess_vec[2] = temperature_leaf_guess;
211 }

213 std::vector<std::vector<double>> ed_c4_leaf_photosynthesis2::get_initial_guesses() const
214 {
215  // Get the default guess to use as a starting point
216  std::vector<double> default_guess(3);
217  get_default(default_guess);
218 
219  // Extract the values
220  double const assimilation_net_guess = default_guess[0];
221  double const conductance_stomatal_h2o_guess = default_guess[1];
222  double const temperature_leaf_guess = default_guess[2];
223 
224  std::string message = "Initial guess calculated by ed_c4_leaf_photosynthesis2:\n";
225  char buff[128];
226  sprintf(buff, " assimilation_net_guess = %e\n", assimilation_net_guess);
227  message += std::string(buff);
228  sprintf(buff, " conductance_stomatal_h2o_guess = %e\n", conductance_stomatal_h2o_guess);
229  message += std::string(buff);
230  sprintf(buff, " temperature_leaf_guess = %e\n", temperature_leaf_guess);
231  message += std::string(buff);
232 
233  string_vector names = {
234  "assimilation_net",
235  "conductance_stomatal_h2o",
236  "temperature_leaf"};
237 
238  message += std::string("\nOrder of unknown variable names as supplied:\n");
239  message += std::string(" ") + names[0] + std::string("\n");
240  message += std::string(" ") + names[1] + std::string("\n");
241  message += std::string(" ") + names[2] + std::string("\n");
242 
243  std::sort(names.begin(), names.end());
244 
245  message += std::string("\nOrder of unknown variable names after applying std::sort:\n");
246  message += std::string(" ") + names[0] + std::string("\n");
247  message += std::string(" ") + names[1] + std::string("\n");
248  message += std::string(" ") + names[2] + std::string("\n");
249 
250  // We also want to try a few perturbations surrounding this guess.
251  // For net assimilation, we'll try using a percentage of the guess value as an additive perturbation, but also enforce a minumum.
252  // For stomatal conductance, we'll try using a multiplicative factor for the perturbation
253  // For leaf temperature, we'll just use a fixed step
254  double const perturbation_fraction_an = 0.1; // dimensionless
255  double const min_an_perturbation = 1e-9; // mol / m^2 / s
256  double const an_perturbation = std::max(perturbation_fraction_an * fabs(assimilation_net_guess), min_an_perturbation); // mol / m^2 / s
257  double const perturbation_fraction_gs = 0.1; // dimensionless
258  double const tl_perturbation = 5.0; // degrees C
259 
260  // For An and Gs, we'll add and subtract the perturbation from each guess `num_pts`
261  // times, i.e., we'll try 2 * num_pts + 1 guesses for each parameter. For Tl, we'll
262  // fix num_pts to be 1. So in total, we will try 3 * (2 * num_pts + 1)^2 guesses.
263  int const num_pts = 2;
264  int const num_pts_tl = 1;
265 
266  std::vector<std::vector<double>> input_guesses;
267 
268  message += "\nGuesses from `ed_c4_leaf_photosynthesis2`:\n";
269 
270  for (int i = -num_pts; i <= num_pts; ++i) {
271  double an_value = assimilation_net_guess + an_perturbation * i;
272  for (int j = -num_pts; j <= num_pts; ++j) {
273  double gs_value = conductance_stomatal_h2o_guess * pow(perturbation_fraction_gs, j);
274  for (int k = -num_pts_tl; k <= num_pts_tl; ++k) {
275  double tl_value = temperature_leaf_guess + tl_perturbation * k;
276  input_guesses.push_back({an_value, gs_value, tl_value});
277  sprintf(buff, " an = %e, gs = %e, tl = %e\n", an_value, gs_value, tl_value);
278  message += std::string(buff);
279  }
280  }
281  }
282 
283  message += "\n";
284  if (eclp2_print) {
285  Rprintf(message.c_str());
286  }
287 
288  return input_guesses;
289 }
