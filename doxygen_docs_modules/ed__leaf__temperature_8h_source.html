140 {
141  // Determine the energy used for evaporating water from the leaf's interior surfaces
142  double const molar_mass_water = 18.0e-3; // kg / mol
143  double const evapotranspiration_rate = conductance_stomatal_h2o *
144  (mole_fraction_h2o_intercellular - mole_fraction_h2o_leaf_surface); // mol / m^2 / s
145  double const et_energy = evapotranspiration_rate * molar_mass_water * latent_heat_vaporization_of_water; // J / m^2 / s
146 
147  // Apply energy balance to calculate the sensible heat loss
148  double const sensible_heat_loss = solar_energy_absorbed_leaf - long_wave_energy_loss_leaf - et_energy; // J / m^2 / s
149 
150  // Use the sensible heat loss to calculate the leaf temperature
151  double const molar_mass_air = 28.97e-3; // kg / mol
152  double const delta_t = sensible_heat_loss /
153  specific_heat_of_air /
154  molar_mass_air /
155  conductance_boundary_h2o; // degrees C
156  double const temperature_leaf = temperature_air + delta_t; // degrees C
157
