202 void ed_multilayer_canopy_properties::run() const
203 {
204  // Calculate PAR levels throughout the canopy. Note that sunML specifies that its
205  // `Idir` and `Idiff` arguments should have [micromol / m^2 / s] units. However,
206  // we can leave these quantities as energy flux densities with [J / m^2 / s] units
207  // because the two units are simply related by a multiplicative conversion factor
208  // and the output of sunML is linear with respect to `Idir` and `Idiff`.
209  struct Light_profile par_profile = sunML(par_incident_direct, par_incident_diffuse,
210  lai, nlayers, cosine_zenith_angle, kd, chil, heightf);
211 
212  // Calculate relative humidity levels throughout the canopy
213  double relative_humidity_profile[nlayers];
214  RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile.
215 
216  // Calculate windspeed throughout the canopy
217  double wind_speed_profile[nlayers];
218  WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile.
219 
220  // Calculate leaf nitrogen throughout the canopy
221  double leafN_profile[nlayers];
222  LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile.
223 
224  // Update layer-dependent outputs
225  //
226  // par_profile.direct_irradiance and par_profile.diffuse_irradiance represent
227  // incident PAR energy flux densities in units of J / m^2 / s. However, the
228  // Collatz model expects a quantum flux in units of mol / m^2 / s. So we use
229  // the `par_energy_content` quantity to convert from J to micromol, and then
230  // an additional factor of 1e6 micromol / mol is required.
231  //
232  // par_profile.total_irradiance represents some kind of wonky "average" incident
233  // PAR flux on the leaves, but the evapotranspiration code expects a value for
234  // total absorbed solar energy (including NIR). To convert, we use the method from
235  // the Evapotrans2 code. First we use the `par_energy_fraction_of_sunlight` quantity
236  // to determine the total amount of solar energy incident on the leaf (including NIR).
237  // Then we determine the amount of light absorbed by the leaf using the following factor:
238  // (1 - LeafReflectance - tau) / (1 - tau) for LeafReflectance = tau = 0.2. This factor
239  // evaluates to a numeric value of (1 - 0.2 - 0.2) / (1 - 0.2) = 0.75.
240  for (int i = 0; i < nlayers; ++i) {
241  update(sunlit_fraction_ops[i], par_profile.sunlit_fraction[i]);
242  update(shaded_fraction_ops[i], par_profile.shaded_fraction[i]);
243  update(height_ops[i], par_profile.height[i]);
244  update(sunlit_incident_par_ops[i], par_profile.direct_irradiance[i]);
245  update(sunlit_collatz_PAR_flux_ops[i], par_profile.direct_irradiance[i] / par_energy_content * 1e-6);
246  update(incident_scattered_par_ops[i], par_profile.scattered_irradiance[i]);
247  update(shaded_incident_par_ops[i], par_profile.diffuse_irradiance[i]);
248  update(shaded_collatz_PAR_flux_ops[i], par_profile.diffuse_irradiance[i] / par_energy_content * 1e-6);
249  update(incident_average_par_ops[i], par_profile.total_irradiance[i]);
250  update(solar_energy_absorbed_leaf_ops[i], par_profile.total_irradiance[i] / par_energy_fraction_of_sunlight * 0.75);
251  update(rh_ops[i], relative_humidity_profile[i]);
252  update(windspeed_ops[i], wind_speed_profile[i]);
253  update(LeafN_ops[i], leafN_profile[i]);
254  }
255 }
