208 void ed_multilayer_canopy_properties::run() const
209 {
210  // Calculate PAR levels throughout the canopy. Note that sunML specifies that its
211  // `Idir` and `Idiff` arguments should have [micromol / m^2 / s] units. However,
212  // we can leave these quantities as energy flux densities with [J / m^2 / s] units
213  // because the two units are simply related by a multiplicative conversion factor
214  // and the output of sunML is linear with respect to `Idir` and `Idiff`.
215  struct Light_profile par_profile = sunML(
216  par_incident_direct, par_incident_diffuse, lai, nlayers,
217  cosine_zenith_angle, kd, chil, absorptivity_par, heightf,
218  par_energy_content, par_energy_fraction, leaf_transmittance,
219  leaf_reflectance);
220 
221  // Calculate relative humidity levels throughout the canopy
222  double relative_humidity_profile[nlayers];
223  RHprof(rh, nlayers, relative_humidity_profile); // Modifies relative_humidity_profile.
224 
225  // Calculate windspeed throughout the canopy
226  double wind_speed_profile[nlayers];
227  WINDprof(windspeed, lai, nlayers, wind_speed_profile); // Modifies wind_speed_profile.
228 
229  // Calculate leaf nitrogen throughout the canopy
230  double leafN_profile[nlayers];
231  LNprof(LeafN, lai, nlayers, kpLN, leafN_profile); // Modifies leafN_profile.
232 
233  // Update layer-dependent outputs
234  //
235  // par_profile.sunlit_incident_ppfd and par_profile.shaded_incident_ppfd represent
236  // incident PAR energy flux densities in units of J / m^2 / s. However, the
237  // Collatz model expects a quantum flux in units of mol / m^2 / s. So we use
238  // the `par_energy_content` quantity to convert from J to micromol, and then
239  // an additional factor of 1e6 micromol / mol is required.
240  //
241  // par_profile.average_incident_ppfd represents some kind of wonky "average" incident
242  // PAR flux on the leaves, but the evapotranspiration code expects a value for
243  // total absorbed solar energy (including NIR). To convert, we use the method from
244  // the Evapotrans2 code. First we use the `par_energy_fraction` quantity
245  // to determine the total amount of solar energy incident on the leaf (including NIR).
246  // Then we determine the amount of light absorbed by the leaf using the following factor:
247  // (1 - LeafReflectance - tau) / (1 - tau)
248  double const absorb_factor = (1 - leaf_reflectance - leaf_transmittance) / (1 - leaf_transmittance);
249  for (int i = 0; i < nlayers; ++i) {
250  update(sunlit_fraction_ops[i], par_profile.sunlit_fraction[i]);
251  update(shaded_fraction_ops[i], par_profile.shaded_fraction[i]);
252  update(height_ops[i], par_profile.height[i]);
253  update(sunlit_incident_par_ops[i], par_profile.sunlit_incident_ppfd[i]);
254  update(sunlit_collatz_PAR_flux_ops[i], par_profile.sunlit_incident_ppfd[i] / par_energy_content * 1e-6);
255  update(incident_scattered_par_ops[i], par_profile.incident_ppfd_scattered[i]);
256  update(shaded_incident_par_ops[i], par_profile.shaded_incident_ppfd[i]);
257  update(shaded_collatz_PAR_flux_ops[i], par_profile.shaded_incident_ppfd[i] / par_energy_content * 1e-6);
258  update(incident_average_par_ops[i], par_profile.average_incident_ppfd[i]);
259  update(solar_energy_absorbed_leaf_ops[i], par_profile.average_incident_ppfd[i] / par_energy_fraction * absorb_factor);
260  update(rh_ops[i], relative_humidity_profile[i]);
261  update(windspeed_ops[i], wind_speed_profile[i]);
262  update(LeafN_ops[i], leafN_profile[i]);
263  }
264 }
