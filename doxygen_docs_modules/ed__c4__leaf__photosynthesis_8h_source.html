201  // Get a crappy estimate for an assimilation rate using
202  // a simplified version of the collatz model.
203  //
204  // Here we make the following assumptions, which are
205  // unlikely to be true but may nevertheless help choose
206  // a good starting guess:
207  // (1) the leaf is at 25 C
208  // (i.e., we don't apply any temperature factors)
209  // (2) the CO2 concentration inside the leaf is equal to the atmospheric value
210  // (i.e., Ci = Ca)
211  // (3) gross assimilation is simply the smallest of the CO2, rubisco, and light limited rates
212  // (i.e., no quadratic mixing)
213  const double assimilation_carbon_limited = collatz_k * mole_fraction_co2_atmosphere; // mol / m^2 / s
214  const double assimilation_rubisco_limited = collatz_vmax; // mol / m^2 / s
215  const double assimilation_light_limited = collatz_alpha * collatz_PAR_flux; // mol / m^2 / s
216  const double assimilation_gross = std::min(assimilation_carbon_limited, std::min(assimilation_rubisco_limited, assimilation_light_limited)); // mol / m^2 / s
217  const double assimilation_net_estimate = assimilation_gross - collatz_rd; // mol / m^2 / s
218 
219  // Get a crappy estimation for stomatal conductance using the
220  // Ball-Berry model.
221  //
222  // Here we make the following assumptions, which are
223  // unlikely to be true but may nevertheless help choose
224  // a good starting guess:
225  // (1) the CO2 concentration at the leaf surface is equal to the atmospheric value
226  // (i.e., Cs = Ca)
227  // (2) the relative humidity at the leaf surface is equal to the atmospheric value
228  // (i.e., RHs = RHa)
229  // (3) net assimilation is given by the estimate calculated above
230  const double ball_berry_index = std::max(0.0, assimilation_net_estimate) * relative_humidity_atmosphere / mole_fraction_co2_atmosphere; // mol / m^2 / s
231  const double ball_berry_conductance = ball_berry_intercept + ball_berry_slope * ball_berry_index; // mol / m^2 / s
232  const double conductance_stomatal_h2o_estimate = conductance_stomatal_h2o_min + StomataWS * (ball_berry_conductance - conductance_stomatal_h2o_min); // mol / m^2 / s
233 
234  // Get a crappy estimate for leaf temperature using the Penman-Monteith model.
235  //
236  // Here we make the following assumptions, which are
237  // unlikely to be true but may nevertheless help choose
238  // a good starting guess:
239  // (1) long wave energy loss from the leaf to its environment is zero
240  // (2) boundary layer conductance is 1.2 mol / m^2 / s, an arbitrary value that
241  // used to be hard-coded into the Ball-Berry model
242  const double latent_heat_vaporization_of_water = TempToLHV(temperature_air); // J / kg
243  const double slope_water_vapor = TempToSFS(temperature_air); // kg / m^3 / K
244  const double saturation_water_vapor_pressure = saturation_vapor_pressure(temperature_air); // Pa
245  const double density_of_dry_air = TempToDdryA(temperature_air); // kg / m^3
246  const double saturation_water_vapor_content = saturation_water_vapor_pressure /
247  physical_constants::ideal_gas_constant /
248  (temperature_air + conversion_constants::celsius_to_kelvin) *
249  physical_constants::molar_mass_of_water; // kg / m^3
250  const double vapor_density_deficit = saturation_water_vapor_content * (1 - relative_humidity_atmosphere); // Pa
251  const double psychrometric_parameter = density_of_dry_air *
252  specific_heat_of_air /
253  latent_heat_vaporization_of_water; // kg / m^3 / K
254  const double Tak = temperature_air + conversion_constants::celsius_to_kelvin; // Kelvin
255  const double volume_per_mol = physical_constants::ideal_gas_constant * Tak / atmospheric_pressure; // m^3 / mol
256 
257  const double total_available_energy = solar_energy_absorbed_leaf; // J / m^2 / s
258  const double gc = conductance_stomatal_h2o_estimate * volume_per_mol; // m / s
259  const double ga = 1.2 * volume_per_mol; // m / s
260  const double delta_t_numerator = total_available_energy * (1 / ga + 1 / gc) - latent_heat_vaporization_of_water * vapor_density_deficit;
261  const double delta_t_denomenator = latent_heat_vaporization_of_water * (slope_water_vapor + psychrometric_parameter * (1 + ga / gc));
262  const double delta_t = delta_t_numerator / delta_t_denomenator; // deg. C
263  const double temperature_leaf_estimate = temperature_air + delta_t; // deg. C
264 
265  // Get a crappy estimate for free boundary layer conductance using the Nikolov model.
266  //
267  // Here we make the following assumptions, which are
268  // unlikely to be true but may nevertheless help choose
269  // a good starting guess:
270  // (1) the leaf temperature is the value estimated above
271  // (2) the relative humidity at the leaf surface is equal to the atmospheric value
272  const double Tlk = temperature_leaf_estimate + conversion_constants::celsius_to_kelvin; // Kelvin
273  const double mole_fraction_h2o_atmosphere = relative_humidity_atmosphere * saturation_water_vapor_pressure / atmospheric_pressure; // dimensionless
274  const double mole_fraction_h2o_leaf_surface = mole_fraction_h2o_atmosphere;
275  const double Tvdiff = Tlk / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_leaf_surface) -
276  Tak / (1.0 - nikolov::Tvdiff_factor * mole_fraction_h2o_atmosphere); // deg. C or K
277  const double gbv_free_estimate = nikolov_ce *
278  pow(Tlk, nikolov::temperature_exponent) *
279  pow((Tlk + nikolov::temperature_offset) / atmospheric_pressure, 0.5) *
280  pow(fabs(Tvdiff) / leafwidth, 0.25) /
281  volume_per_mol; // mol / m^2 / s
