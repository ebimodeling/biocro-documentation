127 void ed_collatz_c4_assimilation::do_operation() const
128 {
129  // Calculate the temperature factor
130  const double temperature_factor = pow(*collatz_q10_ip, (*temperature_leaf_ip - 25.0) / 10.0);
131 
132  // Calculate the carbon-limited assimilation rate
133  const double co2_rate_constant = *collatz_k_ip * temperature_factor;
134  const double assimilation_carbon_limited = co2_rate_constant * *mole_fraction_co2_intercellular_ip;
135 
136  // Calculate the rubisco-limited assimilation rate
137  const double rubisco_capacity_base = *collatz_vmax_ip * temperature_factor;
138  const double rubisco_low_temp_inhibition = 1 + exp(0.3 * (*collatz_rubisco_temperature_lower_ip - *temperature_leaf_ip));
139  const double rubisco_high_temp_inhibition = 1 + exp(0.3 * (*temperature_leaf_ip - *collatz_rubisco_temperature_upper_ip));
140  const double assimilation_rubisco_limited = rubisco_capacity_base / (rubisco_low_temp_inhibition * rubisco_high_temp_inhibition);
141 
142  // Calculate the light-limited assimilation rate
143  const double assimilation_light_limited = *collatz_alpha_ip * *collatz_PAR_flux_ip;
144 
145  // Use the mixing parameter theta to determine the gross assimilation rate limited by rubisco and light
146  const double M_a = *collatz_theta_ip;
147  const double M_b = -1.0 * (assimilation_rubisco_limited + assimilation_light_limited);
148  const double M_c = assimilation_rubisco_limited * assimilation_light_limited;
149  const double M_root_term = M_b * M_b - 4 * M_a * M_c;
150 
151  double M;
152  if (M_a == 0) {
153  M = -M_c / M_b;
154  } else {
155  const double M_plus = (-M_b + sqrt(M_root_term)) / (2 * M_a);
156  const double M_minus = (-M_b - sqrt(M_root_term)) / (2 * M_a);
157  M = std::min(M_plus, M_minus);
158  }
159 
160  // Use the mixing parameter beta to determine the gross assimilation rate limited by carbon, rubisco, and light
161  const double A_a = *collatz_beta_ip;
162  const double A_b = -1.0 * (M + assimilation_carbon_limited);
163  const double A_c = M * assimilation_carbon_limited;
164  const double A_root_term = A_b * A_b - 4 * A_a * A_c;
165 
166  double A;
167  if (A_a == 0) {
168  A = -A_c / A_b;
169  } else {
170  const double A_plus = (-A_b + sqrt(A_root_term)) / (2 * A_a);
171  const double A_minus = (-A_b - sqrt(A_root_term)) / (2 * A_a);
172  A = std::min(A_plus, A_minus);
173  }
174 
175  // Calculate the temperature-dependent leaf respiration
176  const double respiration_base = *collatz_rd_ip * temperature_factor;
177  const double respiration_high_temp_inhibition = 1 + exp(0.3 * (*temperature_leaf_ip - 55.0));
178  const double respiration = respiration_base / respiration_high_temp_inhibition;
179 
180  // Calculate net assimilation rate
181  const double assimilation_net = A - respiration;
182 
183  // Apply a possible reduction accounting for the effect of water stress on the stomata
184  const double adjusted_assimilation_net = assimilation_net * *assimilation_adjustment_factor_WS_ip;
185

187  char buff[128];

190  message += std::string(buff);
191

193  message += std::string(buff);
194

196  message += std::string(buff);
197 
198  sprintf(buff, " rubisco_low_temp_inhibition = %e\n", rubisco_low_temp_inhibition);
199  message += std::string(buff);
200 
201  sprintf(buff, " rubisco_high_temp_inhibition = %e\n", rubisco_high_temp_inhibition);
202  message += std::string(buff);
203 
204  sprintf(buff, " assimilation_rubisco_limited = %e\n", assimilation_rubisco_limited);
205  message += std::string(buff);
206 
207  sprintf(buff, " respiration_base = %e\n", respiration_base);
208  message += std::string(buff);
209 
210  sprintf(buff, " respiration_high_temp_inhibition = %e\n", respiration_high_temp_inhibition);
211  message += std::string(buff);
212 
213  sprintf(buff, " assimilation_net = %e\n", assimilation_net);
214  message += std::string(buff);
215 
216  sprintf(buff, " adjusted_assimilation_net = %e\n", adjusted_assimilation_net);
217  message += std::string(buff);
218 
219  message += std::string("\n");
220  if (collatz_print) {
221  Rprintf(message.c_str());
222  }
223 
224  // Check for error conditions
225  std::map<std::string, bool> errors_to_check = {
226  {"the quadratic M_a and M_b terms cannot both be zero", fabs(M_a) < calculation_constants::eps_zero && fabs(M_b) < calculation_constants::eps_zero}, // divide by zero
227  {"the quadratic M_root_term cannot be negative for M_a != 0", fabs(M_a) > calculation_constants::eps_zero && M_root_term < 0}, // imaginary sqrt
228  {"the quadratic A_a and A_b terms cannot both be zero", fabs(A_a) < calculation_constants::eps_zero && fabs(A_b) < calculation_constants::eps_zero}, // divide by zero
229  {"the quadratic A_root_term cannot be negative for A_a != 0", fabs(A_a) > calculation_constants::eps_zero && A_root_term < 0} // imaginary sqrt
230  };
231 
232  check_error_conditions(errors_to_check, get_name());
233 
234  // Update outputs
235  update(collatz_KT_op, co2_rate_constant);
236  update(collatz_RT_op, respiration);
237  update(collatz_VT_op, assimilation_rubisco_limited);
238  update(assimilation_carbon_limited_op, assimilation_carbon_limited);
239  update(assimilation_rubisco_limited_op, assimilation_rubisco_limited);
240  update(assimilation_light_limited_op, assimilation_light_limited);
241  update(assimilation_gross_op, A);
242  update(respiration_op, respiration);
243  update(assimilation_net_op, adjusted_assimilation_net);
244 }
