143 void multilayer_canopy_integrator::run() const
144 {
145  double const LAIc = lai / nlayers;
146  double canopy_assimilation_rate = 0;
147  double canopy_transpiration_rate = 0;
148  double canopy_conductance = 0;
149  double GrossAssim = 0;
150 
151  // Integrate assimilation, transpiration, and conductance throughout the
152  // canopy
153  for (int i = 0; i < nlayers; ++i) {
154  double const sunlit_lai = *sunlit_fraction_ips[i] * LAIc;
155  double const shaded_lai = *shaded_fraction_ips[i] * LAIc;
156 
157  canopy_assimilation_rate += *sunlit_Assim_ips[i] * sunlit_lai +
158  *shaded_Assim_ips[i] * shaded_lai;
159 
160  canopy_transpiration_rate += *sunlit_TransR_ips[i] * sunlit_lai +
161  *shaded_TransR_ips[i] * shaded_lai;
162 
163  canopy_conductance += *sunlit_Gs_ips[i] * sunlit_lai +
164  *shaded_Gs_ips[i] * shaded_lai;
165 
166  GrossAssim += *sunlit_GrossAssim_ips[i] * sunlit_lai +
167  *shaded_GrossAssim_ips[i] * shaded_lai;
168  }
169 
170  // Modify net assimilation to account for respiration
171  // Note: this was originally only done for the C3 canopy
172  // Note: it seems like this should not be necessary since the assimilation
173  // model includes respiration
174  canopy_assimilation_rate *= (1.0 - growth_respiration_fraction);
175 
176  // For assimilation, we need to convert micromol / m^2 / s into
177  // Mg / ha / hr, assuming that all carbon is converted into biomass in the
178  // form of glucose (C6H12O6), i.e., six assimilated CO2 molecules contribute
179  // one glucose molecule. Using the molar mass of glucose in kg / mol, the
180  // conversion can be accomplished with the following factor:
181  // (1 glucose / 6 CO2) * (3600 s / hr) * (1e-6 mol / micromol) *
182  // (1e-3 Mg / kg) * (1e4 m^2 / ha)
183  // = 6e-3 s * mol * Mg * m^2 / (hr * micromol * kg * ha)
184  const double cf = physical_constants::molar_mass_of_glucose * 6e-3; // (Mg / ha / hr) / (micromol / m^2 / s)
185 
186  // For transpiration, we need to convert mmol / m^2 / s into Mg / ha / hr
187  // using the molar mass of water in kg / mol, which can be accomplished by
188  // the following conversion factor:
189  // (3600 s / hr) * (1e-3 mol / mmol) * (1e-3 Mg / kg) * (1e4 m^2 / ha)
190  // = 36 s * mol * Mg * m^2 / (hr * mmol * kg * ha)
191  const double cf2 = physical_constants::molar_mass_of_water * 36; // (Mg / ha / hr) / (mmol / m^2 / s)
192 
193  update(canopy_assimilation_rate_op, canopy_assimilation_rate * cf);
194 
195  update(GrossAssim_op, GrossAssim * cf);
196 
197  update(canopy_transpiration_rate_op, canopy_transpiration_rate * cf2);
198 
199  update(canopy_conductance_op, canopy_conductance);
200 }
