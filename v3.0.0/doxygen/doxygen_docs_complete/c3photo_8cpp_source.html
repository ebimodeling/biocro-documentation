43 // Temperature corrections are from the following sources:
44 // - Bernacchi et al. (2003) Plant, Cell and Environment, 26(9), 1419-1430.
45 // https://doi.org/10.1046/j.0016-8025.2003.01050.x
46 // - Bernacchi et al. (2001) Plant, Cell and Environment, 24(2), 253-259.
47 // https://doi.org/10.1111/j.1365-3040.2001.00668.x
48 // Note: Values in Dubois and Bernacchi are incorrect.
49 double const Kc = arrhenius_exponential(38.05, 79.43e3, Tleaf_K); // micromol / mol
50 double const Ko = arrhenius_exponential(20.30, 36.38e3, Tleaf_K); // mmol / mol
51 double const Gstar = arrhenius_exponential(19.02, 37.83e3, Tleaf_K); // micromol / mol
52 double const Vcmax = Vcmax0 * arrhenius_exponential(26.35, 65.33e3, Tleaf_K); // micromol / m^2 / s
53 double const Jmax = Jmax0 * arrhenius_exponential(17.57, 43.54e3, Tleaf_K); // micromol / m^2 / s
54 double const Rd = Rd0 * arrhenius_exponential(18.72, 46.39e3, Tleaf_K); // micromol / m^2 / s
55
56 double const theta = thet + 0.018 * Tleaf - 3.7e-4 * pow(Tleaf, 2); // dimensionless
57
58 // Light limited
59 double const dark_adapted_phi_PSII =
60 0.352 + 0.022 * Tleaf - 3.4 * pow(Tleaf, 2) / 1e4; // dimensionless (Bernacchi et al. (2003))
61
62 // The variable that we call `I2` here has been described as "the useful
63 // light absorbed by photosystem II" (S. von Caemmerer (2002)) and "the
64 // maximum fraction of incident quanta that could be utilized in electron
65 // transport" (Bernacchi et al. (2003)). Here we calculate its value using
66 // Equation 3 from Bernacchi et al. (2003), except that we have replaced the
67 // factor `Q * alpha_leaf` (the product of the incident PPFD `Q` and the
68 // leaf absorptance) with the absorbed PPFD, as this is clearly the intended
69 // meaning of the `Q * alpha_leaf` factor. See also Equation 8 from the
70 // original FvCB paper, where `J` (equivalent to our `I2`) is proportional
71 // to the absorbed PPFD rather than the incident PPFD.
72 double const I2 =
73 absorbed_ppfd * dark_adapted_phi_PSII * beta_PSII; // micromol / m^2 / s
74
75 double const J =
76 (Jmax + I2 - sqrt(pow(Jmax + I2, 2) - 4.0 * theta * I2 * Jmax)) /
77 (2.0 * theta); // micromol / m^2 / s
78
79 double const Oi = O2 * solo(Tleaf); // mmol / mol
80

88 // 243, 687-698. https://doi.org/10.1007/s00425-015-2436-8
89 //
90 // In Yang et al., the equation in the caption of Figure 7 calculates the
91 // maximum rate of TPU utilization, but here we need the rate relative to
92 // its value at 25 degrees C (as shown in the figure itself). Using the
93 // equation, the rate at 25 degrees C can be found to have the value
94 // 306.742, so here we normalize the equation by this value.
95 double const TPU_c = 25.5; // dimensionless (fitted constant)
96 double const Ha = 62.99e3; // J / mol (enthalpy of activation)

99 double const R = ideal_gas_constant; // J / K / mol (ideal gas constant)
100 double const top = Tleaf_K * arrhenius_exponential(TPU_c, Ha, Tleaf_K); // dimensionless
101 double const bot = 1.0 + arrhenius_exponential(S / R, Hd, Tleaf_K); // dimensionless
102 double TPU_rate_multiplier = (top / bot) / 306.742; // dimensionless

109

112 double const b1_adj = StomWS * b1;

130
131 // The net CO2 assimilation is the smaller of the biochemistry-limited
132 // and conductance-limited rates. This will prevent the calculated Ci
133 // value from ever being < 0. This seems to be an important restriction
134 // to prevent numerical errors during the convergence loop, but does not
135 // actually limit the net assimilation rate if the loop converges.
136 an_conductance =
137 conductance_limited_assim(Ca, gbw, Gs); // micromol / m^2 / s

154
155 BB_res = ball_berry_gs(
156 co2_assimilation_rate * 1e-6,
157 Ca * 1e-6,
158 RH,
159 b0_adj,
160 b1_adj,
161 gbw,
162 Tleaf,
163 Tambient);

166

168 // and stomata
169 Ci_pa = Ca_pa - AP * (co2_assimilation_rate * 1e-6) *
170 (dr_boundary / gbw + dr_stomata / Gs); // Pa
