26void penman_monteith_leaf_temperature::do_operation() const
27{
28 // From Thornley and Johnson 1990. pg 418. equation 14.11e
29
30 // TODO: This is for about 20 degrees C at 100000 Pa. Change it to use the
31 // model state. (1 * R * temperature) / pressure
32 double constexpr volume_of_one_mole_of_air = 24.39e-3; // m^3 / mol
33
34 double const gc = leaf_stomatal_conductance * 1e-3 * volume_of_one_mole_of_air; // m / s
35
36 double const delta_t_num =
37 leaf_net_irradiance * (1 / leaf_boundary_layer_conductance + 1 / gc) -
38 latent_heat_vaporization_of_water * vapor_density_deficit;
39
40 double const delta_t_denom =
41 latent_heat_vaporization_of_water *
42 (slope_water_vapor + psychrometric_parameter * (1 + leaf_boundary_layer_conductance / gc));
43
44 double const delta_t = delta_t_num / delta_t_denom; // K. It is also degrees C because it's a change in temperature.
45
46 double leaf_temperature = temp + delta_t;
47
48 // Update the output quantity list
49 update(leaf_temperature_op, leaf_temperature);
50}
