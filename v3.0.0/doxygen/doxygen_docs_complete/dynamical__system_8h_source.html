296template <typename time_type>
297void dynamical_system::update_drivers(time_type time_index)
298{
299 size_t const max_index{this->get_ntimes() - 1};
300
301 if (time_index > static_cast<time_type>(max_index)) {
302 throw std::logic_error(
303 "The value of time_index (" + std::to_string(time_index) +
304 ")\nviolates the preconditions for\n" +
305 "'void update_drivers(time_type time_index)',\n" +
306 "which require time_index to be less than or equal to\n" +
307 "the maximum index for a \ndriver variable (" +
308 std::to_string(max_index) + ").");
309 } else {
310 // Because of the checks above, we can safely use
311 // std::vector::operator[] rather than std::vector::at
312 for (auto x : driver_quantity_ptr_pairs) {
313 *(x.first) = (*(x.second))[time_index];
314 }
315 }
316}

337template <class vector_type>
338void dynamical_system::run_differential_modules(vector_type& dxdt)
339{
340 // Reset the derivatives of the differential quantities
341 for (auto& x : differential_quantity_derivatives) {
342 x.second = 0.0;
343 }
344
345 // Run the modules
346 run_module_list(differential_modules);
347
348 // Store the output in the derivative vector
349 for (size_t i = 0; i < dxdt.size(); i++) {
350 dxdt[i] = *(differential_quantity_ptr_pairs[i].second) * (*timestep_ptr);
351 }
352}

368struct push_back_state_and_time {

371 vector<vector_type>& states;
372 vector<double>& times;

374 double threshold = 0;
375 double threshold_increment = 0.02;

380 push_back_state_and_time(

390 // Operation
391 void operator()(vector_type const& x, double t)

393 // Add to the message, if required
394 if (t >= max_time || t / max_time >= threshold) {

398 std::to_string(100.0 * t / max_time) +

401 string(" microseconds\n");
402 threshold += threshold_increment;
403 }

405 // Store the new values
406 states.push_back(x);
407 times.push_back(t);

415template <typename vector_type, typename time_type>
416state_vector_map get_results_from_system(
417 std::shared_ptr<dynamical_system> sys,
418 const vector<vector_type>& x_vec,
419 const vector<time_type>& times)
420{
421 state_vector_map results;
422
423 // Get the keys needed for results
424 string_vector output_param_names = sys->get_output_quantity_names();
425
426 // Initialize all result values to all-zero vectors of the correct length
427 vector<double> temporary(x_vec.size());
428 for (const string& p : output_param_names) {
429 results[p] = temporary;
430 }
431
432 // Add an entry for ncalls (hopefully there will be a better way to do this
433 // someday)
434 std::fill(temporary.begin(), temporary.end(), sys->get_ncalls());
435 results["ncalls"] = temporary;
436
437 // Get pointers to the output quantities
438 vector<const double*> output_param_ptrs =
439 sys->get_quantity_access_ptrs(output_param_names);
440
441 // Store the data
442 for (size_t i = 0; i < x_vec.size(); ++i) {
443 sys->update_all_quantities(x_vec[i], times[i]);
444 for (size_t j = 0; j < output_param_names.size(); ++j) {
445 (results[output_param_names[j]])[i] = *(output_param_ptrs[j]);
446 }
447 }
448
449 return results;
450}
