123void dynamical_system::update_drivers(double time_index)
124{
125 size_t const max_index{this->get_ntimes() - 1};
126
127 // Find two closest surrounding integers within the time bounds. Sometimes
128 // roundoff errors may cause `ceil(time_index)` to be out-of-bounds, so we
129 // restrict it to a maximum value.
130 size_t const t1{static_cast<size_t>(std::floor(time_index))};
131
132 size_t const t2{std::min(
133 static_cast<size_t>(std::ceil(time_index)), max_index)};
134
135 if (t1 > max_index) {
136 throw std::logic_error(
137 "The value of time_index (" + std::to_string(time_index) +
138 ")\nviolates the preconditions for\n" +
139 "'void update_drivers(double time_index)',\n" +
140 "which require floor(time_index) to be less than or equal to\n" +
141 "the maximum index for a \ndriver variable (" +
142 std::to_string(max_index) + ").");
143 } else if (t1 == t2) {
144 // No need to interpolate in this case; we can just use the method for
145 // integral times.
146 this->update_drivers(t1);
147 } else {
148 // Use linear interpolation to find the value of each driver at
149 // time_index. Because of the checks above, we can use
150 // std::vector::operator[] rather than std::vector::at.
151 for (const auto& x : driver_quantity_ptr_pairs) {
152 auto value_at_t1 = (*(x.second))[t1];
153 auto value_at_t2 = (*(x.second))[t2];
154 auto value_at_time_index =
155 value_at_t1 + (time_index - t1) * (value_at_t2 - value_at_t1);
156
157 *(x.first) = value_at_time_index;
158 }
159 }
160}
