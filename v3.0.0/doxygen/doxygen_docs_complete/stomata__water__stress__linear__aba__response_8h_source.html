78 double const slope = 1.0 / (soil_field_capacity - soil_wilting_point);
79 double const intercept = 1.0 - soil_field_capacity * slope;
80 double const aba_effect = exp(soil_aba_concentration / aba_influence_coefficient); // dimensionless. A value in the interval (0, 1] that will reduce the slope of the Ball-Berry model.
81
82 // Update the output quantity list
83 update(StomataWS_op, std::min(std::max(slope * soil_water_content + intercept, 1e-10), 1.0));
84 update(b1_op, max_b1 * aba_effect);
