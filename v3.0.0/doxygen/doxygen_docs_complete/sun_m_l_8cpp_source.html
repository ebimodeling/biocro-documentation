188Light_profile sunML(
189 double ambient_ppfd_beam, // micromol / (m^2 beam) / s
190 double ambient_ppfd_diffuse, // micromol / m^2 / s
191 double lai, // dimensionless from m^2 / m^2
192 int nlayers, // dimensionless
193 double cosine_zenith_angle, // dimensionless
194 double kd, // dimensionless
195 double chil, // dimensionless from m^2 / m^2
196 double absorptivity, // dimensionless from mol / mol
197 double heightf, // m^-1 from m^2 leaf / m^2 ground / m height
198 double par_energy_content, // J / micromol
199 double par_energy_fraction, // dimensionless
200 double leaf_transmittance, // dimensionless
201 double leaf_reflectance // dimensionless
202)
203{
204 if (nlayers < 1 || nlayers > MAXLAY) {
205 throw std::out_of_range("nlayers must be at least 1 but no more than MAXLAY.");
206 }
207 if (cosine_zenith_angle > 1 || cosine_zenith_angle < -1) {
208 throw std::out_of_range("cosine_zenith_angle must be between -1 and 1.");
209 }
210 if (kd > 1 || kd < 0) {
211 throw std::out_of_range("kd must be between 0 and 1.");
212 }
213 if (chil < 0) {
214 throw std::out_of_range("chil must be non-negative.");
215 }
216 if (absorptivity > 1 || absorptivity < 0) {
217 throw std::out_of_range("absorptivity must be between 0 and 1.");
218 }
219 if (heightf <= 0) {
220 throw std::out_of_range("heightf must greater than zero.");
221 }
222
223 // Calculate the leaf shape factor for an ellipsoidal leaf angle
224 // distribution using the equation from page 251 of Campbell & Norman
225 // (1998). We will use this value as `k`, the canopy extinction coefficient
226 // for photosynthetically active radiation throughout the canopy. This
227 // quantity represents the ratio of horizontal area to total area for leaves
228 // in the canopy and is therefore dimensionless from
229 // (m^2 ground) / (m^2 leaf).
230 double zenith_angle = acos(cosine_zenith_angle); // radians
231 double k0 = sqrt(pow(chil, 2) + pow(tan(zenith_angle), 2));
232 double k1 = chil + 1.744 * pow((chil + 1.182), -0.733);
233 double k = k0 / k1; // dimensionless
234
235 double lai_per_layer = lai / nlayers;
236
237 // Calculate the fraction of direct radiation that passes through the canopy
238 // using Equation 15.1. Note that this is equivalent to the fraction of
239 // ground area below the canopy that is exposed to direct sunlight. Note
240 // that if the sun is at or below the horizon, no part of the soil is
241 // sunlit; this corresponds to the case where cosine_zenith_angle is close
242 // to or below zero.
243 double canopy_direct_transmission_fraction =
244 cosine_zenith_angle <= 1E-10 ? 0.0 : exp(-k * lai); // dimensionless
245
246 // Calculate the ambient direct PPFD through a surface parallel to the ground
247 const double ambient_ppfd_beam_ground = ambient_ppfd_beam * cosine_zenith_angle; // micromol / (m^2 ground) / s
248
249 // Calculate the ambient direct PPFD through a unit area of leaf surface
250 double ambient_ppfd_beam_leaf = ambient_ppfd_beam_ground * k; // micromol / (m^2 leaf) / s
251
252 // Start to fill in the light profile values
253 Light_profile light_profile;
254 light_profile.canopy_direct_transmission_fraction = canopy_direct_transmission_fraction;
255
256 // Fill in the layer-dependent light profile values
257 for (int i = 0; i < nlayers; ++i) {
258 // Get the cumulative LAI for this layer, which represents the total
259 // leaf area above this layer
260 const double cumulative_lai = lai_per_layer * (i + 0.5);
261
262 // Calculate the amount of PPFD scattered out of the direct beam using
263 // Equations 15.6 and 15.1 from Campbell & Norman (1998), following
264 // example 15.2. This is a diffuse flux density representing the flux
265 // through any surface.
266 const double scattered_ppfd =
267 ambient_ppfd_beam_ground * (exp(-k * sqrt(absorptivity) * cumulative_lai) -
268 exp(-k * cumulative_lai)); // micromol / m^2 / s
269
270 // Calculate the total flux of diffuse photosynthetically active light
271 // in this layer by combining the scattered PPFD with the ambient
272 // diffuse PPFD. Here we use Equation 15.6 with `alpha` = 1 and
273 // `kbe(phi)` = kd.
274 double diffuse_ppfd =
275 ambient_ppfd_diffuse * exp(-kd * cumulative_lai) + scattered_ppfd; // micromol / m^2 / s
276
277 // Calculate the fraction of sunlit and shaded leaves in this canopy
278 // layer using Equation 15.21.
279 const double Ls = (1 - exp(-k * lai_per_layer)) * exp(-k * cumulative_lai) / k; // dimensionless
280 double sunlit_fraction = Ls / lai_per_layer; // dimensionless
281 double shaded_fraction = 1 - sunlit_fraction; // dimensionless
282
283 // Calculate an "average" incident PPFD for the sunlit and shaded leaves
284 // that doesn't seem to be based on a formula from Campbell & Norman
285 // (1998). It's interpreted as a flux density through a unit of leaf
286 // area, but that may not be correct.
287 double average_ppfd =
288 (sunlit_fraction * (ambient_ppfd_beam_leaf + diffuse_ppfd) + shaded_fraction * diffuse_ppfd) *
289 (1 - exp(-k * lai_per_layer)) / k; // micromol / (m^2 leaf) / s
290
291 // For values of cosine_zenith_angle close to or less than 0, in place
292 // of the calculations above, we want to use the limits of the above
293 // expressions as cosine_zenith_angle approaches 0 from the right:
294 if (cosine_zenith_angle <= 1E-10) {
295 ambient_ppfd_beam_leaf = ambient_ppfd_beam / k1;
296 diffuse_ppfd = ambient_ppfd_diffuse * exp(-kd * cumulative_lai);
297 sunlit_fraction = 0;
298 shaded_fraction = 1;
299 average_ppfd = 0;
300 }
301
302 // Store values of incident PPFD
303 light_profile.sunlit_incident_ppfd[i] = ambient_ppfd_beam_leaf + diffuse_ppfd; // micromol / (m^2 leaf) / s
304 light_profile.incident_ppfd_scattered[i] = scattered_ppfd; // micromol / m^2 / s
305 light_profile.shaded_incident_ppfd[i] = diffuse_ppfd; // micromol / (m^2 leaf) / s
306 light_profile.average_incident_ppfd[i] = average_ppfd; // micromol / (m^2 leaf) / s
307 light_profile.sunlit_fraction[i] = sunlit_fraction; // dimensionless from m^2 / m^2
308 light_profile.shaded_fraction[i] = shaded_fraction; // dimensionless from m^2 / m^2
309 light_profile.height[i] = (lai - cumulative_lai) / heightf; // m
310
311 // Store values of absorbed PPFD
312 light_profile.sunlit_absorbed_ppfd[i] =
313 thin_layer_absorption(
314 leaf_reflectance,
315 leaf_transmittance,
316 ambient_ppfd_beam_leaf + diffuse_ppfd); // micromol / m^2 / s
317
318 light_profile.shaded_absorbed_ppfd[i] =
319 thin_layer_absorption(
320 leaf_reflectance,
321 leaf_transmittance,
322 diffuse_ppfd); // micromol / m^2 / s
323
324 // Store values of absorbed solar energy (including PAR and NIR)
325 light_profile.sunlit_absorbed_shortwave[i] =
326 absorbed_shortwave_from_incident_ppfd(
327 ambient_ppfd_beam_leaf + diffuse_ppfd,
328 par_energy_content,
329 par_energy_fraction,
330 leaf_reflectance,
331 leaf_transmittance); // J / (m^2 leaf) / s
332
333 light_profile.shaded_absorbed_shortwave[i] =
334 absorbed_shortwave_from_incident_ppfd(
335 diffuse_ppfd,
336 par_energy_content,
337 par_energy_fraction,
338 leaf_reflectance,
339 leaf_transmittance); // J / (m^2 leaf) / s
340
341 light_profile.average_absorbed_shortwave[i] =
342 absorbed_shortwave_from_incident_ppfd(
343 average_ppfd,
344 par_energy_content,
345 par_energy_fraction,
346 leaf_reflectance,
347 leaf_transmittance); // J / (m^2 leaf) / s
348 }
349 return light_profile;
