8/*
9 * !!!!! IMPORTANT NOTE !!!!!:
10 * This is a C++ interface to the C library in R. C++ supports exceptions, but C
11 * does not. R's C library requires you to explicitly PROTECT() and UNPROTECT()
12 * memory for garbage collection. If care is not taken, and an exception is
13 * thrown after memory has been PROTECTed, but before it has been UNPROTECTed,
14 * UNPROTECT will not be called.
15 *
16 * Every effort should be made to not use exception-throwing functions between
17 * PROTECT and UNPROTECT pairs. If an exception-throwing function is used
18 * between pairs, a means of ensuring that UNPROTECT is called when exceptions
19 * are thrown is required.
20 *
21 * The ScopeGuard pattern could be used which would store protect_count
22 * internally, and call UNPROTECT(protect_count) in the destructor. It would be
23 * used as follows:
24 *
25 * guard = ProtectGuard();
26 * PROTECT(something);
27 * guard.increment_protect_count(1);
28 *
29 * ProtectGuard is an implementation of the ScopeGuard pattern. Its
30 * increment_protect_count method should be be called immediately after each set
31 * of PROTECT calls using the appropriate number of items to process. The
32 * ProtectGuard destructor will be called when the function exists, regardless
33 * of whether exceptions were thrown.
34 *
35 * For reference: the following methods are guaranteed to not throw exceptions:
36 * - std::unordered_map::begin()
37 * - std::unordered_map::end()
38 * - std::unordered_map::size()
39 * - std::unordered_map::operator[()
40 * - std::vector::begin()
41 * - std::vector::end()
42 * - std::vector::size()
43 * - std::vector::operator[()
44 */

59{
60 SEXP names = Rf_getAttrib(list, R_NamesSymbol);
61 size_t n = Rf_length(list);
62 state_vector_map m;
63 m.reserve(n);
64 for (size_t i = 0; i < n; ++i) {
65 size_t p = Rf_length(VECTOR_ELT(list, i));
66 vector<double> temporary;
67 temporary.reserve(p);
68 for (size_t j = 0; j < p; ++j) {
69 temporary.emplace_back(REAL(VECTOR_ELT(list, i))[j]);
70 }
71 m.insert(state_vector_map::value_type(
72 CHAR(STRING_ELT(names, i)), temporary));
73 }

181SEXP list_from_module_info(
182 std::string const& module_name,
183 string_vector const& module_inputs,
184 string_vector const& module_outputs,
185 bool const& is_differential,
186 bool const& requires_euler_ode_solver,
187 std::string const& creation_error_message)
188{
189 // Module inputs and outputs (these do not require an UNPROTECT here)
190 SEXP input_vec = PROTECT(r_string_vector_from_vector(module_inputs));
191 SEXP output_vec = PROTECT(r_string_vector_from_vector(module_outputs));
192
193 // Module name
194 SEXP name = PROTECT(Rf_allocVector(STRSXP, 1));
195 SET_STRING_ELT(name, 0, Rf_mkChar(module_name.c_str()));
196
197 // Module type
198 SEXP type = PROTECT(Rf_allocVector(STRSXP, 1));
199 if (is_differential) {
200 SET_STRING_ELT(type, 0, Rf_mkChar("differential"));
201 } else {
202 SET_STRING_ELT(type, 0, Rf_mkChar("direct"));
203 }
204
205 // Euler requirement
206 SEXP requires_euler = PROTECT(Rf_allocVector(STRSXP, 1));
207 string euler_msg;
208 if (requires_euler_ode_solver) {
209 euler_msg = "requires a fixed-step Euler ode_solver";
210 } else {
211 euler_msg = "does not require a fixed-step Euler ode_solver";
212 }
213 SET_STRING_ELT(requires_euler, 0, Rf_mkChar(euler_msg.c_str()));
214
215 // Error message from module creation
216 SEXP error_message = PROTECT(Rf_allocVector(STRSXP, 1));
217 SET_STRING_ELT(error_message, 0, Rf_mkChar(creation_error_message.c_str()));
218
219 // The module info list
220 SEXP info_names = PROTECT(Rf_allocVector(STRSXP, 6));
221 SET_STRING_ELT(info_names, 0, Rf_mkChar("module_name"));
222 SET_STRING_ELT(info_names, 1, Rf_mkChar("inputs"));
223 SET_STRING_ELT(info_names, 2, Rf_mkChar("outputs"));
224 SET_STRING_ELT(info_names, 3, Rf_mkChar("type"));
225 SET_STRING_ELT(info_names, 4, Rf_mkChar("euler_requirement"));
226 SET_STRING_ELT(info_names, 5, Rf_mkChar("creation_error_message"));
227
228 SEXP info_list = PROTECT(Rf_allocVector(VECSXP, 6));
229 SET_VECTOR_ELT(info_list, 0, name);
230 SET_VECTOR_ELT(info_list, 1, input_vec);
231 SET_VECTOR_ELT(info_list, 2, output_vec);
232 SET_VECTOR_ELT(info_list, 3, type);
233 SET_VECTOR_ELT(info_list, 4, requires_euler);
234 SET_VECTOR_ELT(info_list, 5, error_message);
235
236 Rf_setAttrib(info_list, R_NamesSymbol, info_names);
237
238 UNPROTECT(8);
239
240 return info_list;
241}
