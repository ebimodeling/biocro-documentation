142void partitioning_coefficient_logistic::do_operation() const
143{
144 // Determine partitioning coefficients using multinomial logistic equations
145 // from Osborne et al., 2015 JULES-crop https://doi.org/10.5194/gmd-8-1139-2015
146
147 // denominator term for kRoot, kStem, kLeaf, and kGrain
148 double kDenom = exp(alphaRoot + betaRoot * DVI) +
149 exp(alphaStem + betaStem * DVI) +
150 exp(alphaLeaf + betaLeaf * DVI) +
151 exp(alphaShell + betaShell * DVI) + 1.0; // dimensionless
152
153 double kRoot = kcoeff(alphaRoot, betaRoot, DVI, kDenom); // dimensionless
154 double kStem = kcoeff(alphaStem, betaStem, DVI, kDenom); // dimensionless
155 double kLeaf = kcoeff(alphaLeaf, betaLeaf, DVI, kDenom); // dimensionless
156 double kShell = kcoeff(alphaShell, betaShell, DVI, kDenom); // dimensionless
157 double kGrain = 1.0 / kDenom; // dimensionless
158
159 // Give option for rhizome to contribute to growth during the emergence stage,
160 // kRhizome_emr is an input parameter and should be non-positive.
161 // To ignore rhizome, set kRhizome_emr to 0 in input parameter file, and
162 // adjust initial leaf, stem, and root biomasses accordingly.
163 double kRhizome{DVI < 0 ? kRhizome_emr : 0}; // dimensionless
164
165 // Update the output quantities
166 update(kRoot_op, kRoot); // dimensionless
167 update(kStem_op, kStem); // dimensionless
168 update(kLeaf_op, kLeaf); // dimensionless
169 update(kShell_op, kShell); // dimensionless
170 update(kGrain_op, kGrain); // dimensionless
171 update(kRhizome_op, kRhizome); // dimensionless
172}
173
174double kcoeff(double alpha, double beta, double DVI, double denom)
175{
176 return exp(alpha + beta * DVI) / denom; // dimensionless
177}
