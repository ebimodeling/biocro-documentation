130 // Initialize
131 FvCB_outputs result;
132
133 // Calculate rates
134 if (Ci == 0.0) {
135 // RuBP-saturated net assimilation rate when Ci is 0
136 double Ac0 =
137 -Gstar * Vcmax / (Kc * (1 + Oi / Ko)) - Rd; // micromol / m^2 / s
138
139 // RuBP-regeneration-limited net assimilation when C is 0
140 double Aj0 =
141 -J / (2.0 * electrons_per_oxygenation) - Rd; // micromol / m^2 / s
142
143 // Store results; note that TPU cannot be limiting when
144 // Ci < Gstar * (1 + 3 * alpha_TPU) and that An = max(Ac, Aj) when
145 // Ci < Gstar.
146 result.An = std::max(Ac0, Aj0); // micromol / m^2 / s
147 result.Ac = Ac0; // micromol / m^2 / s
148 result.Aj = Aj0; // micromol / m^2 / s
149 result.Ap = inf; // micromol / m^2 / s
150 result.Vc = 0.0; // micromol / m^2 / s
151 result.Wc = 0.0; // micromol / m^2 / s
152 result.Wj = 0.0; // micromol / m^2 / s
153 result.Wp = inf; // micromol / m^2 / s
154
155 } else {
156 // RuBP-saturated carboxylation rate
157 double Wc = Vcmax * Ci /
158 (Ci + Kc * (1.0 + Oi / Ko)); // micromol / m^2 / s
159
160 // RuBP-regeneration-limited carboxylation rate (micromol / m^2 / s)
161 double Wj = J * Ci /
162 (electrons_per_carboxylation * Ci +
163 2.0 * electrons_per_oxygenation * Gstar);
164
165 // Triose-phosphate-utilization-limited carboxylation rate. There is an
166 // asymptote at Ci = Gstar * (1 + 3 * alpha_TPU), and TPU cannot limit
167 // the carboxylation rate for values of Ci below this asymptote. A
168 // simple way to handle this is to make Wp infinite for
169 // Ci <= Gstar * (1 + 3 * alpha_TPU), so that it is never limiting in
170 // this case.
171 double Wp =
172 Ci > Gstar * (1.0 + 3.0 * alpha_TPU)
173 ? 3.0 * TPU * Ci / (Ci - Gstar * (1.0 + 3.0 * alpha_TPU))
174 : inf; // micromol / m^2 / s
175
176 // Assimilated carbon per carboxylation
177 double a_per_c = (1.0 - Gstar / Ci); // dimensionless
178
179 // Limiting carboxylation rate
180 double Vc = std::min(Wc, std::min(Wj, Wp)); // micromol / m^2 / s
181
182 // Store results
183 result.An = a_per_c * Vc - Rd; // micromol / m^2 / s
184 result.Ac = a_per_c * Wc - Rd; // micromol / m^2 / s
185 result.Aj = a_per_c * Wj - Rd; // micromol / m^2 / s
186 result.Ap = a_per_c * Wp - Rd; // micromol / m^2 / s
187 result.Vc = Vc; // micromol / m^2 / s
188 result.Wc = Wc; // micromol / m^2 / s
189 result.Wj = Wj; // micromol / m^2 / s
190 result.Wp = Wp; // micromol / m^2 / s
191 }
192
193 return result;
