155void grimm_soybean_flowering_calculator::do_operation() const
156{
157 // Calculate the night length
158 const double night_length = 24.0 - day_length; // hours
159
160 // Calculate the normalized growth rate based on the day of year, night
161 // length, temperature, and growth stage
162 double rate_normalized = 0.0; // dimensionless
163 double temperature_rate_normalized = 0.0; // dimensionless
164 double photoperiod_rate_normalized = 0.0; // dimensionless
165 if (time < sowing_time) {
166 // The plant has not been sowed yet, so it can't grow
167 rate_normalized = 0.0;
168 } else if (grimm_physiological_age < grimm_juvenile_pd_threshold) {
169 // The plant is still in the juvenile (photoperiod-insensitive) phase
170 if (temp < grimm_juvenile_T0) {
171 // The temperature is too low for the plant to grow
172 rate_normalized = 0.0;
173 } else if (temp < grimm_juvenile_T1) {
174 // There is a linear increase from 0 to 1 as temp changes from T0 to
175 // T1
176 rate_normalized = (temp - grimm_juvenile_T0) /
177 (grimm_juvenile_T1 - grimm_juvenile_T0);
178 } else if (temp < grimm_juvenile_T2) {
179 // We are in the flat region at maximum growth
180 rate_normalized = 1.0;
181 } else if (temp < grimm_juvenile_T3) {
182 // There is a linear decrease from 1 to 0 as temp changes from T2 to
183 // T3
184 rate_normalized = 1.0 - (temp - grimm_juvenile_T2) /
185 (grimm_juvenile_T3 - grimm_juvenile_T2);
186 } else {
187 // The temperature is too high for the plant to grow
188 rate_normalized = 0.0;
189 }
190 } else {
191 // The plant has entered the photoperiod-sensitive phase
192 if (temp < grimm_T_min) {
193 // The temperature is too low for the plant to develop
194 temperature_rate_normalized = 0.0;
195 } else if (temp < grimm_T_opt) {
196 // There is a linear increase from 0 to 1 as temp changes from T_min
197 // to T_opt
198 temperature_rate_normalized = (temp - grimm_T_min) /
199 (grimm_T_opt - grimm_T_min);
200 } else {
201 // The development rate is maxed out at 1
202 temperature_rate_normalized = 1.0;
203 }
204 if (night_length < grimm_N_min) {
205 // The night is too short for the plant to develop
206 photoperiod_rate_normalized = 0.0;
207 } else if (night_length < grimm_N_opt) {
208 // There is a linear increase from 0 to 1 as night_length changes
209 // from N_min to N_opt
210 photoperiod_rate_normalized = (night_length - grimm_N_min) /
211 (grimm_N_opt - grimm_N_min);
212 } else {
213 // The development rate is maxed out at 1
214 photoperiod_rate_normalized = 1.0;
215 }
216 rate_normalized = temperature_rate_normalized * photoperiod_rate_normalized;
217 }
218
219 // Decide if the plant is flowering
220 const bool is_flowering =
221 (grimm_physiological_age > grimm_juvenile_pd_threshold + grimm_flowering_threshold);
222
223 // Scale the rates to the maximum value, which is 1 physiological day per
224 // day, equivalent to 1 / 24 physiological days per hour
225 const double rate_per_hour = rate_normalized / 24.0;
226 const double temperature_rate_per_hour = temperature_rate_normalized / 24.0;
227 const double photoperiod_rate_per_hour = photoperiod_rate_normalized / 24.0;
228
229 // Update the output quantity list
230 update(grimm_rate_op, rate_per_hour);
231 update(grimm_rate_temperature_op, temperature_rate_per_hour);
232 update(grimm_rate_photoperiod_op, photoperiod_rate_per_hour);
233 update(grimm_flowering_op, is_flowering);
234}
