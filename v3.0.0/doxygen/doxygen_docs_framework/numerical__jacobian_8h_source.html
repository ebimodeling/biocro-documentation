104template <typename equation_ptr_type,
105 typename vector_type,
106 typename time_type,
107 typename matrix_type>
108void calculate_jacobian(
109 equation_ptr_type const& equation_ptr,
110 vector_type const& x,
111 time_type t,
112 vector_type const& f_current,
113 matrix_type& jacobi)
114{
115 size_t n = x.size();
116
117 // Make a vector to store the perturbed f(x,t)
118 vector_type f_perturbed(n);
119
120 // Perturb each element x_i of the input vector to find df_j(x,t)/dx_i,
121 // which is stored at jacobi(j,i)
122 double h{};
123 double abs_xi{};
124 double min_h = pow(calculation_constants::eps_deriv, 2);
125 vector_type x_perturbed = x;
126
127 for (size_t i = 0; i < n; i++) {
128 // Determine the step size h by taking a fraction of the absolute value
129 // of x[i]. Set a minimum value for h in case x[i] is zero or very
130 // small, since we don't want to divide by zero.
131 abs_xi = fabs(x[i]);
132 h = std::max(abs_xi * calculation_constants::eps_deriv, min_h);
133
134 // Ensure that the step size h is close to this value but is exactly
135 // representable in the computer architecture.
136 double volatile temp = x[i] + h;
137 h = temp - x[i];
138
139 // Calculate the new function value by adding h to the ith element of x
140 // and evaluating f, storing its value in f_perturbed.
141 x_perturbed[i] = x[i] + h;
142 evaluate_equations(equation_ptr, x_perturbed, t, f_perturbed);
143
144 // Store the results in the Jacobian matrix
145 for (size_t j = 0; j < n; j++) {
146 jacobi(j, i) = (f_perturbed[j] - f_current[j]) / h;
147 }
148
149 // Reset the ith differential quantity
150 x_perturbed[i] = x[i];
151 }
152}

261template <typename equation_ptr_type,
262 typename time_type,
263 typename vector_type>
264void calculate_time_derivative(
265 equation_ptr_type const& equation_ptr,
266 time_type max_time,
267 vector_type const& x,
268 time_type t,
269 vector_type const& f_current,
270 vector_type& dfdt)
271{
272 size_t n = x.size();
273
274 // Make a vector to store the perturbed f(x,t)
275 vector_type f_perturbed(n);
276
277 // Determine the step size h by taking a fraction of t. Set a minimum value
278 // for h in case t is zero or very small, since we don't want to divide by
279 // zero.
280 double h = std::max(t * calculation_constants::eps_deriv,
281 pow(calculation_constants::eps_deriv, 2));
282
283 // Use a forward step whenever possible.
284 if (t + h <= max_time) {
285 // Ensure that the step size h is close to this value but is exactly
286 // representable in the computer architecture.
287 double volatile temp = t + h;
288 h = temp - t;
289
290 // Calculate the new function value by adding h to t and evaluating f,
291 // storing its value in f_perturbed.
292 evaluate_equations(equation_ptr, x, t + h, f_perturbed);
293
294 // Store the results in the derivative vector
295 for (size_t j = 0; j < n; j++) {
296 dfdt[j] = (f_perturbed[j] - f_current[j]) / h;
297 }
298 } else {
299 // Ensure that the step size h is close to this value but is exactly
300 // representable in the computer architecture.
301 double volatile temp = t - h;
302 h = t - temp;
303
304 // Calculate the new function value by subtracting h from t and
305 // evaluating f, storing its value in f_perturbed.
306 evaluate_equations(equation_ptr, x, t - h, f_perturbed);
307
308 // Store the results in the derivative vector
309 for (size_t j = 0; j < n; j++) {
310 dfdt[j] = (f_current[j] - f_perturbed[j]) / h;
311 }
312 }
313}
