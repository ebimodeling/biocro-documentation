73// Store some information that will be useful to any type of boost ode_solver, and then call the private do_boost_integrate method
74template <class state_type>
75state_vector_map boost_ode_solver<state_type>::do_integrate(std::shared_ptr<dynamical_system> sys)
76{
77 // Update and/or reset the stored objects
78 sys->get_differential_quantities(state);
79 state_vec.clear();
80 time_vec.clear();
81 observer_message = std::string("");
82
83 // Make an observer
84 push_back_state_and_time<state_type> observer(state_vec, time_vec, sys->get_ntimes() - 1.0, observer_message);
85
86 // Make a system caller
87 dynamical_system_caller syscall{sys};
88
89 // integrate the system (modifies state_vec and time_vec via the observer)
90 do_boost_integrate(syscall, observer);
91
92 // Return the results
93 return get_results_from_system(sys, state_vec, time_vec);
94}

96// Run integrate_const using stored information and the supplied stepper
97template <class state_type>
98template <class stepper_type>
99void boost_ode_solver<state_type>::run_integrate_const(stepper_type stepper, dynamical_system_caller syscall, push_back_state_and_time<state_type> observer)
100{
101 try {
102 nsteps = boost::numeric::odeint::integrate_const(
103 stepper,
104 syscall,
105 state,
106 0.0,
107 syscall.get_ntimes() - 1.0,
108 get_output_step_size(),
109 observer,
110 boost::numeric::odeint::max_step_checker(get_adaptive_max_steps()));
111 boost_error_string.clear();
112 } catch (std::exception& e) {
113 // Store the error message and let the ode_solver return the partial results
114 nsteps = 0;
115 boost_error_string = std::string(e.what());
116 }
117}

190 double const rel_err = this->get_adaptive_rel_error_tol();

193 auto stepper = boost::numeric::odeint::make_controlled<error_stepper_type>(abs_err, rel_err);
194
195 // Run integrate_const
