40// The name of this function must be `R_init_NAME`, where `NAME` must exactly
41// match the name of the package
42void attribute_visible R_init_BioCro(DllInfo* info)
43{
44 // We only use .Call in our R code, so we only need to register call methods
45 R_registerRoutines(info, NULL, callMethods, NULL, NULL);
46
47 // Don't allow searching; i.e., only allow registered symbols to be called
48 // from R code
49 R_useDynamicSymbols(info, FALSE);
50
51 // Only allow R objects (not character strings) when using .Call from R
52 R_forceSymbols(info, TRUE);
53}
