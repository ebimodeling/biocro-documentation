78void water_vapor_properties_from_air_temperature::do_operation() const
79{
80 // Collect input quantities and make calculations
81
82 double density_of_dry_air = TempToDdryA(temp); // kg / m^3
83 double latent_heat_vaporization_of_water = TempToLHV(temp); // J / kg
84 double saturation_water_vapor_pressure = saturation_vapor_pressure(temp); // Pa
85
86 // Convert saturation water vapor pressure to vapor density using the ideal
87 // gas law. This is approximately right for temperatures what won't kill
88 // plants.
89 double saturation_water_vapor_content =
90 saturation_water_vapor_pressure / physical_constants::ideal_gas_constant /
91 (temp + conversion_constants::celsius_to_kelvin) *
92 physical_constants::molar_mass_of_water; // kg / m^3
93
94 double vapor_density_deficit = saturation_water_vapor_content * (1 - rh); // kg / m^3
95
96 double psychrometric_parameter =
97 density_of_dry_air * specific_heat_of_air /
98 latent_heat_vaporization_of_water; // kg / m^3 / K
99
100 // Update the output quantity list
101 update(latent_heat_vaporization_of_water_op, latent_heat_vaporization_of_water); // J / kg
102 update(slope_water_vapor_op, TempToSFS(temp)); // kg / m^3 / K
103 update(saturation_water_vapor_pressure_op, saturation_water_vapor_pressure); // Pa
104 update(water_vapor_pressure_op, saturation_water_vapor_pressure * rh); // Pa
105 update(vapor_density_deficit_op, vapor_density_deficit); // kg / m^3
106 update(psychrometric_parameter_op, psychrometric_parameter); // kg / m^3 / K
107}
