109void thermal_time_development_rate_calculator::do_operation() const
110{
111 // Calculate the development_rate
112 double development_rate; // day^-1
113 double temp_diff = temp - tbase; // degrees C
114 temp_diff = (temp_diff > 0) ? temp_diff : 0; // if temp < tbase, temp_diff = 0
115
116 if (time < sowing_time) {
117 // The seeds haven't been sown yet, so no development should occur
118 development_rate = 0; // day^-1
119 } else if (DVI < -1) {
120 // error, DVI out of bounds, this should never occur unless initial DVI
121 // state is less than -1.
122 development_rate = 0;
123 } else if (DVI < 0) {
124 // 1. Sowing to emergence
125 development_rate = temp_diff / TTemr; // day^-1
126
127 } else if (DVI < 1) {
128 // 2. Vegetative stages
129 development_rate = temp_diff / TTveg; // day^-1
130
131 } else {
132 // 3. Reproductive Stages
133 development_rate = temp_diff / TTrep; // day^-1
134 }
135
136 double development_rate_per_hour = development_rate / 24.0; // hour^-1
137
138 // Update the output quantity list
139 update(development_rate_per_hour_op, development_rate_per_hour);
140}
