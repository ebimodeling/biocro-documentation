98void magic_clock::do_operation() const
99{
101 // Collect inputs and make calculations //
103
104 using math_constants::pi;
105
106 // Get the current values of the dawn and dusk kicks
107 const double dawn_kick = *dawn_kick_ip;
108 const double dusk_kick = *dusk_kick_ip;
109
110 // Get the current state of the dawn tracking oscillator
111 const double dawn_b = *dawn_b_ip;
112 const double dawn_a = *dawn_a_ip;
113
114 // Get the current state of the dusk tracking oscillator
115 const double dusk_b = *dusk_b_ip;
116 const double dusk_a = *dusk_a_ip;
117
118 // Get the current state of the reference oscillator
119 const double ref_b = *ref_b_ip;
120 const double ref_a = *ref_a_ip;
121
122 // Define the constants for the tracking oscillators
123 const double natural_period = *clock_period_ip; // Natural period in hours
124 const double natural_freq = 2.0 * pi / natural_period; // Corresponding angular frequency in radians per hour
125 const double gamma = *clock_gamma_ip;
126 const double r_0 = *clock_r0_ip;
127
128 // Calculate the friction terms for the tracking oscillators
129 const double dawn_friction = gamma * (r_0 - sqrt(dawn_a * dawn_a + dawn_b * dawn_b));
130 const double dusk_friction = gamma * (r_0 - sqrt(dusk_a * dusk_a + dusk_b * dusk_b));
131 const double ref_friction = gamma * (r_0 - sqrt(ref_a * ref_a + ref_b * ref_b));
132
134 // Update the output quantity list //
136
137 // The dawn tracking oscillator is driven by a small kick at dawn
138 // The kick is created by taking the night tracker value during the day,
139 // which is just a short decay portion
140 update(dawn_b_op, natural_freq * dawn_a);
141 update(dawn_a_op, dawn_friction * dawn_a - natural_freq * dawn_b + dawn_kick);
142
143 // The dusk tracking oscillator is driven by a small kick at dusk
144 // The kick is created by taking the day tracker value during the night,
145 // which is just a short decay portion
146 update(dusk_b_op, natural_freq * dusk_a);
147 update(dusk_a_op, dusk_friction * dusk_a - natural_freq * dusk_b + dusk_kick);
148
149 // The reference oscillator is not driven
150 update(ref_b_op, natural_freq * ref_a);
151 update(ref_a_op, ref_friction * ref_a - natural_freq * ref_b);
152}
